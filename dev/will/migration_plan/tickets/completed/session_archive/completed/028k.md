### Implementation
Modify `GenerateHandoff()` signature:

```go
func GenerateHandoff(cfg *HandoffConfig, metrics *SessionMetrics) (*Handoff, *HandoffMetrics, error) {
	startTime := time.Now()

	// Existing generation logic...
	handoff := &Handoff{
		SessionID:        metrics.SessionID,
		Timestamp:        time.Now().Unix(),
		SchemaVersion:    HandoffSchemaVersion,
		SessionMetrics:   metrics,
		PendingLearnings: pendingLearnings,
		Violations:       violations,
		GitInfo:          gitInfo,
	}

	// Write JSONL
	if err := writeHandoffJSONL(cfg.HandoffPath, handoff); err != nil {
		return nil, nil, fmt.Errorf("[handoff] Failed to write JSONL: %w", err)
	}

	// Collect metrics
	handoffMetrics := &HandoffMetrics{
		GenerationTimeMs: time.Since(startTime).Milliseconds(),
		SharpEdgeCount:   len(pendingLearnings),
		ViolationCount:   len(violations),
		PatternCount:     countPatterns(violations), // Helper function
	}

	return handoff, handoffMetrics, nil
}
```

### Helper Function
```go
// countPatterns counts unique violation patterns in violations list
func countPatterns(violations []Violation) int {
	patterns := make(map[string]bool)
	for _, v := range violations {
		patterns[v.ViolationType] = true
	}
	return len(patterns)
}
```

func TestGenerateHandoff_MetricTimingAccurate(t *testing.T) {
	tmpDir := t.TempDir()
	cfg := &HandoffConfig{
		HandoffPath: filepath.Join(tmpDir, "handoffs.jsonl"),
	}

	metrics := &SessionMetrics{
		SessionID: "timing-test",
		ToolCalls: 5,
		Timestamp: time.Now().Unix(),
	}

	startBefore := time.Now()
	_, handoffMetrics, err := GenerateHandoff(cfg, metrics)
	endAfter := time.Now()

	if err != nil {
		t.Fatalf("Expected no error, got: %v", err)
	}

	expectedMinMs := startBefore.Sub(endAfter).Milliseconds()
	expectedMaxMs := endAfter.Sub(startBefore).Milliseconds()

	if handoffMetrics.GenerationTimeMs < expectedMinMs || handoffMetrics.GenerationTimeMs > expectedMaxMs {
		t.Errorf("Generation time %dms outside expected range [%d, %d]",
			handoffMetrics.GenerationTimeMs, expectedMinMs, expectedMaxMs)
	}
}
