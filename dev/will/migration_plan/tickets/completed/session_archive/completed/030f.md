### ClusterInvocationsByAgent Function
```go
// ClusterInvocationsByAgent groups invocations by agent and calculates aggregate stats.
// Returns map of agent name to aggregated statistics.
func ClusterInvocationsByAgent(invocations []AgentInvocation) map[string]*AgentInvocationStats {
	clusters := make(map[string]*AgentInvocationStats)

	for _, inv := range invocations {
		agent := inv.Agent
		if agent == "" {
			agent = "unknown"
		}

		stats, exists := clusters[agent]
		if !exists {
			stats = &AgentInvocationStats{
				Agent:   agent,
				Samples: []AgentInvocation{},
			}
			clusters[agent] = stats
		}

		stats.TotalCount++
		stats.TotalDurationMs += inv.DurationMs
		stats.TotalInputTokens += inv.InputTokens
		stats.TotalOutputTokens += inv.OutputTokens
		stats.TotalThinkingTokens += inv.ThinkingTokens

		if inv.Success {
			stats.SuccessCount++
		} else {
			stats.FailureCount++
		}

		// Keep first 3 as samples
		if len(stats.Samples) < 3 {
			stats.Samples = append(stats.Samples, inv)
		}
	}

	// Calculate derived metrics
	for _, stats := range clusters {
		if stats.TotalCount > 0 {
			stats.SuccessRate = float64(stats.SuccessCount) / float64(stats.TotalCount)
			stats.AvgDurationMs = stats.TotalDurationMs / int64(stats.TotalCount)
		}
	}

	return clusters
}
```

```go
// ClusterInvocationsByTier groups invocations by model tier.
// Useful for understanding cost distribution across tiers.
func ClusterInvocationsByTier(invocations []AgentInvocation) map[string]*TierInvocationStats {
	clusters := make(map[string]*TierInvocationStats)

	for _, inv := range invocations {
		tier := inv.Tier
		if tier == "" {
			tier = "unknown"
		}

		stats, exists := clusters[tier]
		if !exists {
			stats = &TierInvocationStats{
				Tier:           tier,
				AgentBreakdown: make(map[string]int),
			}
			clusters[tier] = stats
		}

		stats.TotalCount++
		stats.TotalInputTokens += inv.InputTokens
		stats.TotalOutputTokens += inv.OutputTokens

		if inv.Success {
			stats.SuccessCount++
		}

		// Track agent breakdown within tier
		stats.AgentBreakdown[inv.Agent]++
	}

	// Calculate success rates
	for _, stats := range clusters {
		if stats.TotalCount > 0 {
			stats.SuccessRate = float64(stats.SuccessCount) / float64(stats.TotalCount)
		}
	}

	return clusters
}
```

### Ranking Functions
```go
// AgentRanking represents an agent with a sortable metric
type AgentRanking struct {
	Agent  string  `json:"agent"`
	Metric float64 `json:"metric"`
	Count  int     `json:"count"`
}

// GetTopAgentsByUsage returns agents sorted by invocation count (descending).
func GetTopAgentsByUsage(invocations []AgentInvocation, limit int) []AgentRanking {
	clusters := ClusterInvocationsByAgent(invocations)

	rankings := make([]AgentRanking, 0, len(clusters))
	for agent, stats := range clusters {
		rankings = append(rankings, AgentRanking{
			Agent:  agent,
			Metric: float64(stats.TotalCount),
			Count:  stats.TotalCount,
		})
	}

	// Sort by count descending
	sort.Slice(rankings, func(i, j int) bool {
		return rankings[i].Count > rankings[j].Count
	})

	if limit > 0 && limit < len(rankings) {
		return rankings[:limit]
	}
	return rankings
}

// GetTopAgentsByErrorRate returns agents sorted by error rate (descending).
// Only includes agents with at least minInvocations to avoid noise.
func GetTopAgentsByErrorRate(invocations []AgentInvocation, limit int, minInvocations int) []AgentRanking {
	clusters := ClusterInvocationsByAgent(invocations)

	rankings := make([]AgentRanking, 0, len(clusters))
	for agent, stats := range clusters {
		if stats.TotalCount < minInvocations {
			continue // Skip agents with too few invocations
		}
		errorRate := 1.0 - stats.SuccessRate
		rankings = append(rankings, AgentRanking{
			Agent:  agent,
			Metric: errorRate,
			Count:  stats.TotalCount,
		})
	}

	// Sort by error rate descending
	sort.Slice(rankings, func(i, j int) bool {
		return rankings[i].Metric > rankings[j].Metric
	})

	if limit > 0 && limit < len(rankings) {
		return rankings[:limit]
	}
	return rankings
}

// GetTopAgentsByLatency returns agents sorted by average latency (descending).
func GetTopAgentsByLatency(invocations []AgentInvocation, limit int) []AgentRanking {
	clusters := ClusterInvocationsByAgent(invocations)

	rankings := make([]AgentRanking, 0, len(clusters))
	for agent, stats := range clusters {
		rankings = append(rankings, AgentRanking{
			Agent:  agent,
			Metric: float64(stats.AvgDurationMs),
			Count:  stats.TotalCount,
		})
	}

	// Sort by latency descending
	sort.Slice(rankings, func(i, j int) bool {
		return rankings[i].Metric > rankings[j].Metric
	})

	if limit > 0 && limit < len(rankings) {
		return rankings[:limit]
	}
	return rankings
}
```

```go
func TestClusterInvocationsByAgent_SingleAgent(t *testing.T) {
	invocations := []AgentInvocation{
		{Agent: "python-pro", Success: true, DurationMs: 1000, InputTokens: 500, OutputTokens: 250},
		{Agent: "python-pro", Success: true, DurationMs: 1500, InputTokens: 600, OutputTokens: 300},
		{Agent: "python-pro", Success: false, DurationMs: 500, InputTokens: 200, OutputTokens: 50},
	}

	clusters := ClusterInvocationsByAgent(invocations)

	if len(clusters) != 1 {
		t.Errorf("Expected 1 cluster, got: %d", len(clusters))
	}

	stats := clusters["python-pro"]
	if stats.TotalCount != 3 {
		t.Errorf("Expected TotalCount 3, got: %d", stats.TotalCount)
	}
	if stats.SuccessCount != 2 {
		t.Errorf("Expected SuccessCount 2, got: %d", stats.SuccessCount)
	}
	if stats.FailureCount != 1 {
		t.Errorf("Expected FailureCount 1, got: %d", stats.FailureCount)

		t.Errorf("Expected SuccessRate ~0.67, got: %f", stats.SuccessRate)

}
```
