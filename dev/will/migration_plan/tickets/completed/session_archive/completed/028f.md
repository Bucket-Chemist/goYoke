### Test Script
File: `test/compatibility/context_loading_test.sh`

```bash
#!/bin/bash
set -euo pipefail

# Generate test markdown using Go implementation
TMPDIR=$(mktemp -d)
trap "rm -rf $TMPDIR" EXIT

export GOGENT_PROJECT_DIR="$TMPDIR"
mkdir -p "$TMPDIR/.claude/memory"

# Create mock session event
cat > "$TMPDIR/session.json" <<EOF
{
  "session_id": "test-session",
  "timestamp": 1234567890,
  "hook_event_name": "SessionEnd"
}
EOF

# Run gogent-archive to generate markdown
if ! command -v gogent-archive &>/dev/null; then
    echo "❌ gogent-archive not installed. Run: make install"
    exit 1
fi

echo "Generating Go handoff markdown..."
gogent-archive < "$TMPDIR/session.json" >/dev/null 2>&1 || true

HANDOFF_MD="$TMPDIR/.claude/memory/last-handoff.md"

if [[ ! -f "$HANDOFF_MD" ]]; then
    echo "❌ Go handoff markdown not generated"
    exit 1
fi

echo "✅ Go handoff markdown generated"

# Test if load-routing-context.sh can parse it
if [[ ! -f ~/.claude/hooks/load-routing-context.sh ]]; then
    echo "⚠️  load-routing-context.sh not found, skipping compatibility test"
    exit 0
fi

echo "Testing load-routing-context.sh parsing..."

# Simulate SessionStart hook call (simplified)
# Actual implementation varies - this is a placeholder
if grep -q "Session Metrics" "$HANDOFF_MD"; then
    echo "✅ Markdown contains expected sections"
else
    echo "❌ Markdown missing expected sections"
    exit 1
fi

# Check section headers match bash expectations
EXPECTED_HEADERS=(
    "Session Metrics"
    "Sharp Edges"
    "Routing Violations"
    "Recommended Actions"
)

for header in "${EXPECTED_HEADERS[@]}"; do
    if ! grep -q "## $header" "$HANDOFF_MD"; then
        echo "❌ Missing section: $header"
        exit 1
    fi
done

echo "✅ All expected section headers present"
echo ""
echo "Markdown preview:"
head -20 "$HANDOFF_MD"
```
