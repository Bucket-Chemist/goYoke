# GOgent-036: Implement Failure Tracking in pkg/memory

> **Unified with 036b**: This ticket implements both basic tracking AND composite key
> support (file+error) to prevent false positives from the start.

## Background

**Current State:**
- `pkg/routing/failure.go` provides `DetectFailure()` (from 034)
- No tracking layer exists yet
- Shell script uses file-only tracking → false positives
- Need: Persistent, windowed, composite-key tracking

**Goal:**
Create reusable tracking package that multiple CLIs can use.

## Implementation Intention

Build `pkg/memory/failure_tracking.go` with:
1. Composite key tracking (file + error_type)
2. Sliding time window
3. Configurable thresholds
4. Persistent storage in `~/.gogent/`
5. Graceful degradation (missing files → count=0)

## Implementation Details

### pkg/memory/failure_tracking.go

```go
package memory

import (
	"bufio"
	"encoding/json"
	"fmt"
	"os"
	"path/filepath"
	"time"
)

// FailureKey uniquely identifies a failure pattern
// Uses composite key (file + error type) to prevent false positives
type FailureKey struct {
	FilePath  string
	ErrorType string
	Function  string // Optional - populated by 036c
}

// String returns canonical key representation for logging
func (k FailureKey) String() string {
	if k.Function != "" {
		return fmt.Sprintf("%s:%s:%s", k.FilePath, k.ErrorType, k.Function)
	}
	return fmt.Sprintf("%s:%s", k.FilePath, k.ErrorType)
}

// FailureEntry represents a logged failure event
type FailureEntry struct {
	Timestamp int64  `json:"timestamp"`
	File      string `json:"file"`
	Tool      string `json:"tool"`
	ErrorType string `json:"error_type"`
	Function  string `json:"function,omitempty"` // Optional
}

// FailureTracker tracks consecutive failures within a time window
type FailureTracker struct {
	ErrorLogPath  string
	FailureWindow int // seconds
	MaxFailures   int
}

// DefaultFailureTracker creates a tracker with standard configuration
func DefaultFailureTracker() *FailureTracker {
	homeDir, _ := os.UserHomeDir()
	return &FailureTracker{
		ErrorLogPath:  filepath.Join(homeDir, ".gogent", "failure-tracker.jsonl"),
		FailureWindow: getEnvInt("GOGENT_FAILURE_WINDOW", 300),
		MaxFailures:   getEnvInt("GOGENT_MAX_FAILURES", 3),
	}
}

// LogFailure appends a failure entry to the log
func (ft *FailureTracker) LogFailure(key FailureKey, tool string) error {
	// Ensure directory exists
	if err := os.MkdirAll(filepath.Dir(ft.ErrorLogPath), 0755); err != nil {
		return fmt.Errorf("[failure-tracker] Failed to create log directory: %w", err)
	}

	entry := FailureEntry{
		Timestamp: time.Now().Unix(),
		File:      key.FilePath,
		Tool:      tool,
		ErrorType: key.ErrorType,
		Function:  key.Function,
	}

	data, err := json.Marshal(entry)
	if err != nil {
		return fmt.Errorf("[failure-tracker] Failed to marshal entry: %w", err)
	}

	f, err := os.OpenFile(ft.ErrorLogPath, os.O_APPEND|os.O_CREATE|os.O_WRONLY, 0644)
	if err != nil {
		return fmt.Errorf("[failure-tracker] Failed to open log: %w", err)
	}
	defer f.Close()

	if _, err := f.Write(append(data, '\n')); err != nil {
		return fmt.Errorf("[failure-tracker] Failed to write: %w", err)
	}

	return nil
}

// CountRecentFailures counts failures matching the key within the time window
// Uses composite key: file + error_type (+ function if present)
func (ft *FailureTracker) CountRecentFailures(key FailureKey) (int, error) {
	f, err := os.Open(ft.ErrorLogPath)
	if err != nil {
		if os.IsNotExist(err) {
			return 0, nil // Graceful degradation
		}
		return 0, fmt.Errorf("[failure-tracker] Failed to open error log: %w", err)
	}
	defer f.Close()

	cutoff := time.Now().Unix() - int64(ft.FailureWindow)
	count := 0

	scanner := bufio.NewScanner(f)
	var lines []string
	for scanner.Scan() {
		lines = append(lines, scanner.Text())
		// Keep only last 100 lines in memory
		if len(lines) > 100 {
			lines = lines[1:]
		}
	}

	for _, line := range lines {
		var entry FailureEntry
		if err := json.Unmarshal([]byte(line), &entry); err != nil {
			continue // Skip malformed lines
		}

		// Composite key matching
		if entry.File == key.FilePath &&
			entry.ErrorType == key.ErrorType &&
			entry.Timestamp > cutoff {

			// If function is specified in key, must match
			if key.Function != "" && entry.Function != key.Function {
				continue
			}

			count++
		}
	}

	return count, nil
}

// ClearFailures removes all failures for a given key (optional reset behavior)
func (ft *FailureTracker) ClearFailures(key FailureKey) error {
	f, err := os.Open(ft.ErrorLogPath)
	if err != nil {
		if os.IsNotExist(err) {
			return nil // Nothing to clear
		}
		return fmt.Errorf("[failure-tracker] Failed to open log: %w", err)
	}
	defer f.Close()

	var kept []FailureEntry
	scanner := bufio.NewScanner(f)
	for scanner.Scan() {
		var entry FailureEntry
		if err := json.Unmarshal(scanner.Bytes(), &entry); err != nil {
			continue
		}

		// Keep entries that don't match the key
		if !(entry.File == key.FilePath && entry.ErrorType == key.ErrorType) {
			kept = append(kept, entry)
		}
	}

	// Rewrite log with filtered entries
	temp := ft.ErrorLogPath + ".tmp"
	tempFile, err := os.Create(temp)
	if err != nil {
		return fmt.Errorf("[failure-tracker] Failed to create temp file: %w", err)
	}

	for _, entry := range kept {
		data, _ := json.Marshal(entry)
		tempFile.Write(append(data, '\n'))
	}
	tempFile.Close()

	return os.Rename(temp, ft.ErrorLogPath)
}

func getEnvInt(key string, defaultValue int) int {
	if v := os.Getenv(key); v != "" {
		var i int
		fmt.Sscanf(v, "%d", &i)
		if i > 0 {
			return i
		}
	}
	return defaultValue
}
```

## Test Requirements

### Test: Basic Logging
```go
func TestLogFailure(t *testing.T) {
	tracker := &FailureTracker{
		ErrorLogPath: filepath.Join(t.TempDir(), "test.jsonl"),
	}

	key := FailureKey{FilePath: "test.go", ErrorType: "syntax_error"}
	err := tracker.LogFailure(key, "Edit")

	assert.NoError(t, err)
	assert.FileExists(t, tracker.ErrorLogPath)
}
```

### Test: Composite Key Prevents False Positives
```go
func TestCompositeKeyFiltering(t *testing.T) {
	tracker := setupTracker(t)

	// Log 3 different errors on same file
	tracker.LogFailure(FailureKey{"main.go", "TypeError", ""}, "Bash")
	tracker.LogFailure(FailureKey{"main.go", "ValueError", ""}, "Bash")
	tracker.LogFailure(FailureKey{"main.go", "SyntaxError", ""}, "Bash")

	// Count for each specific error
	count1, _ := tracker.CountRecentFailures(FailureKey{"main.go", "TypeError", ""})
	count2, _ := tracker.CountRecentFailures(FailureKey{"main.go", "ValueError", ""})

	// Each error counted separately
	assert.Equal(t, 1, count1)
	assert.Equal(t, 1, count2)
}
```

### Test: Same Error Triggers Threshold
```go
func TestSameErrorThreshold(t *testing.T) {
	tracker := setupTracker(t)

	key := FailureKey{"main.go", "TypeError", ""}

	// Log same error 3 times
	tracker.LogFailure(key, "Bash")
	tracker.LogFailure(key, "Bash")
	tracker.LogFailure(key, "Bash")

	count, _ := tracker.CountRecentFailures(key)
	assert.Equal(t, 3, count)
}
```

### Test: Time Window Filtering
```go
func TestTimeWindowFiltering(t *testing.T) {
	tracker := &FailureTracker{
		ErrorLogPath:  filepath.Join(t.TempDir(), "test.jsonl"),
		FailureWindow: 60, // 1 minute
	}

	// Manually write old entry (61 seconds ago)
	oldEntry := FailureEntry{
		Timestamp: time.Now().Unix() - 61,
		File:      "test.go",
		ErrorType: "error",
	}
	writeEntry(tracker.ErrorLogPath, oldEntry)

	// Recent entry
	tracker.LogFailure(FailureKey{"test.go", "error", ""}, "Bash")

	count, _ := tracker.CountRecentFailures(FailureKey{"test.go", "error", ""})

	// Only recent entry counted
	assert.Equal(t, 1, count)
}
```

## Why Composite Key Matters

**Without composite key (file-only):**
```
File: main.go
- Line 10: TypeError
- Line 50: ImportError
- Line 100: ValueError
→ 3 failures on main.go → TRIGGERS (false positive)
```

**With composite key (file + error):**
```
File: main.go, Error: TypeError → Count: 1
File: main.go, Error: ImportError → Count: 1
File: main.go, Error: ValueError → Count: 1
→ No trigger (correct)
```
