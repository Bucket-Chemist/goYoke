package integration

import (
	"bytes"
	"encoding/json"
	"os"
	"os/exec"
	"path/filepath"
	"strings"
	"testing"

	"github.com/Bucket-Chemist/GOgent-Fortress/pkg/session"
)

func TestSessionHandoffIntegration_FullWorkflow(t *testing.T) {
	// Skip if binary not built
	binaryPath := "../../bin/gogent-archive"
	if _, err := os.Stat(binaryPath); os.IsNotExist(err) {
		t.Skip("gogent-archive binary not built. Run: make build-archive")
	}

	// Setup temp project
	projectDir := t.TempDir()
	memoryDir := filepath.Join(projectDir, ".claude", "memory")
	os.MkdirAll(memoryDir, 0755)

	// Setup temp runtime dir
	runtimeDir := t.TempDir()
	os.Setenv("XDG_RUNTIME_DIR", runtimeDir)
	defer os.Unsetenv("XDG_RUNTIME_DIR")

	gogentDir := filepath.Join(runtimeDir, "gogent")
	os.MkdirAll(gogentDir, 0755)

	// Pre-populate metrics files
	toolCounter := filepath.Join(gogentDir, "claude-tool-counter-integration.log")
	os.WriteFile(toolCounter, []byte("call1\ncall2\ncall3\ncall4\ncall5\n"), 0644)

	errorLog := filepath.Join(gogentDir, "claude-error-patterns.jsonl")
	os.WriteFile(errorLog, []byte(`{"error":"type1"}
{"error":"type2"}
`), 0644)

	violationsLog := filepath.Join(memoryDir, "routing-violations.jsonl")
	os.WriteFile(violationsLog, []byte(`{"violation":"subagent_mismatch"}
{"violation":"delegation_ceiling"}
{"violation":"model_mismatch"}
`), 0644)

	// Pre-populate learnings
	learningsLog := filepath.Join(memoryDir, "pending-learnings.jsonl")
	os.WriteFile(learningsLog, []byte(`{"file":"test.go","error_type":"parse_error","failure_count":3,"last_occurrence":1234567890}
`), 0644)

	// Create SessionEnd event
	sessionEvent := map[string]interface{}{
		"session_id":      "integration-test-abc123",
		"timestamp":       1234567890,
		"hook_event_name": "SessionEnd",
	}
	eventJSON, _ := json.Marshal(sessionEvent)

	// Invoke gogent-archive
	cmd := exec.Command(binaryPath)
	cmd.Env = append(os.Environ(), "GOGENT_PROJECT_DIR="+projectDir)
	cmd.Stdin = bytes.NewReader(eventJSON)

	var stdout, stderr bytes.Buffer
	cmd.Stdout = &stdout
	cmd.Stderr = &stderr

	if err := cmd.Run(); err != nil {
		t.Fatalf("gogent-archive failed: %v\nStderr: %s", err, stderr.String())
	}

	// Verify confirmation JSON
	var confirmation map[string]interface{}
	if err := json.Unmarshal(stdout.Bytes(), &confirmation); err != nil {
		t.Fatalf("Invalid confirmation JSON: %v\nOutput: %s", err, stdout.String())
	}

	hookOutput := confirmation["hookSpecificOutput"].(map[string]interface{})
	if hookOutput["session_id"] != "integration-test-abc123" {
		t.Errorf("Confirmation missing session_id: %v", hookOutput)
	}

	// Verify handoffs.jsonl created
	handoffJSONL := filepath.Join(memoryDir, "handoffs.jsonl")
	handoffData, err := os.ReadFile(handoffJSONL)
	if err != nil {
		t.Fatalf("handoffs.jsonl not created: %v", err)
	}

	// Parse JSONL
	var handoff session.Handoff
	if err := json.Unmarshal(handoffData, &handoff); err != nil {
		t.Fatalf("Invalid handoff JSONL: %v", err)
	}

	// Verify metrics
	if handoff.Context.Metrics.ToolCalls != 5 {
		t.Errorf("Expected 5 tool calls, got %d", handoff.Context.Metrics.ToolCalls)
	}
	if handoff.Context.Metrics.ErrorsLogged != 2 {
		t.Errorf("Expected 2 errors, got %d", handoff.Context.Metrics.ErrorsLogged)
	}
	if handoff.Context.Metrics.RoutingViolations != 3 {
		t.Errorf("Expected 3 violations, got %d", handoff.Context.Metrics.RoutingViolations)
	}

	// Verify last-handoff.md created
	handoffMD := filepath.Join(memoryDir, "last-handoff.md")
	mdContent, err := os.ReadFile(handoffMD)
	if err != nil {
		t.Fatalf("last-handoff.md not created: %v", err)
	}

	// Verify markdown sections
	requiredSections := []string{
		"## Session Metrics",
		"## Sharp Edges",
		"## Routing Violations",
		"## Recommended Actions",
	}
	for _, section := range requiredSections {
		if !strings.Contains(string(mdContent), section) {
			t.Errorf("Markdown missing section: %s", section)
		}
	}

	// Verify artifacts archived
	archiveDir := filepath.Join(memoryDir, "session-archive")
	if _, err := os.Stat(archiveDir); os.IsNotExist(err) {
		t.Fatal("session-archive/ directory not created")
	}

	// Verify original learnings moved
	if _, err := os.Stat(learningsLog); !os.IsNotExist(err) {
		t.Error("Learnings log should have been moved to archive")
	}

	// Verify original violations moved
	if _, err := os.Stat(violationsLog); !os.IsNotExist(err) {
		t.Error("Violations log should have been moved to archive")
	}

	// Verify archived files exist
	entries, err := os.ReadDir(archiveDir)
	if err != nil {
		t.Fatalf("Failed to read archive dir: %v", err)
	}

	hasLearnings := false
	hasViolations := false
	for _, entry := range entries {
		if strings.HasPrefix(entry.Name(), "learnings-") {
			hasLearnings = true
		}
		if strings.HasPrefix(entry.Name(), "violations-") {
			hasViolations = true
		}
	}

	if !hasLearnings {
		t.Error("Archived learnings file not found")
	}
	if !hasViolations {
		t.Error("Archived violations file not found")
	}
}

func TestSessionHandoffIntegration_MinimalSession(t *testing.T) {
	binaryPath := "../../bin/gogent-archive"
	if _, err := os.Stat(binaryPath); os.IsNotExist(err) {
		t.Skip("gogent-archive binary not built")
	}

	// Test with no artifacts (empty session)
	projectDir := t.TempDir()
	memoryDir := filepath.Join(projectDir, ".claude", "memory")
	os.MkdirAll(memoryDir, 0755)

	runtimeDir := t.TempDir()
	os.Setenv("XDG_RUNTIME_DIR", runtimeDir)
	defer os.Unsetenv("XDG_RUNTIME_DIR")

	sessionEvent := map[string]interface{}{
		"session_id":      "minimal-session",
		"timestamp":       1234567890,
		"hook_event_name": "SessionEnd",
	}
	eventJSON, _ := json.Marshal(sessionEvent)

	cmd := exec.Command(binaryPath)
	cmd.Env = append(os.Environ(), "GOGENT_PROJECT_DIR="+projectDir)
	cmd.Stdin = bytes.NewReader(eventJSON)

	var stdout bytes.Buffer
	cmd.Stdout = &stdout

	// Should not error on missing artifacts
	if err := cmd.Run(); err != nil {
		t.Fatalf("Should handle missing artifacts gracefully: %v", err)
	}

	// Handoff should still be created
	handoffJSONL := filepath.Join(memoryDir, "handoffs.jsonl")
	if _, err := os.Stat(handoffJSONL); os.IsNotExist(err) {
		t.Error("Handoff should be created even with no artifacts")
	}
}

## Why This Matters

Integration tests catch bugs that unit tests miss: file path mismatches, schema incompatibilities, timing issues, environment dependencies. This test validates the entire workflow end-to-end before production deployment.

**CRITICAL**: The 028j expansion added extensive CLI surface area (--version, --help, error handling, STDIN timeout). Without integration tests, these can break in production despite passing unit tests. CLI boundary issues are the #1 cause of deployment failures.

**Production Confidence**: If this test passes, handoff workflow AND CLI subcommands are proven functional.
