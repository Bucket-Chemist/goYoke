### Implementation Pattern
```go
func AnalyzeViolationTrend(violations []routing.Violation) *TrendAnalysis {
	if len(violations) < 2 {
		return &TrendAnalysis{
			Trend:   "insufficient_data",
			Message: "Not enough violations to analyze trend",
		}
	}

	// Sort by timestamp
	sort.Slice(violations, func(i, j int) bool {
		return violations[i].Timestamp < violations[j].Timestamp
	})

	// Find session midpoint
	firstTS := violations[0].Timestamp
	lastTS := violations[len(violations)-1].Timestamp
	midpoint := firstTS + (lastTS-firstTS)/2

	// Count early vs late
	earlyCount := 0
	lateCount := 0

	for _, v := range violations {
		if v.Timestamp < midpoint {
			earlyCount++
		} else {
			lateCount++
		}
	}

	// Determine trend
	analysis := &TrendAnalysis{
		EarlyCount: earlyCount,
		LateCount:  lateCount,
	}

	if lateCount < earlyCount {
		analysis.Trend = "improving"
		reduction := int(float64(earlyCount-lateCount) / float64(earlyCount) * 100)
		analysis.Message = fmt.Sprintf("✅ Improving: %d%% reduction in second half of session", reduction)
	} else if lateCount > earlyCount {
		analysis.Trend = "worsening"
		increase := int(float64(lateCount-earlyCount) / float64(earlyCount) * 100)
		analysis.Message = fmt.Sprintf("⚠️ Worsening: %d%% increase in second half of session", increase)
	} else {
		analysis.Trend = "stable"
		analysis.Message = "Stable: consistent violation rate throughout session"
	}

	return analysis
}
```
