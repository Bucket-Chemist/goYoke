### Implementation Pattern
```go
func GenerateBlockingResponse(edge *SharpEdge, index *SharpEdgeIndex, failureCount int) *HookResponse {
	// Find similar patterns
	matches := FindSimilar(edge, index)

	// Build base message
	baseMsg := fmt.Sprintf(
		"🔴 DEBUGGING LOOP DETECTED (%d failures on %s):\n"+
			"1. STOP current approach\n"+
			"2. Document this sharp edge (auto-logged to pending-learnings.jsonl)\n"+
			"3. Analyze root cause - what assumption might be wrong?\n"+
			"4. Consider escalation to next tier\n"+
			"5. Check sharp-edges.yaml for similar patterns",
		failureCount,
		edge.File,
	)

	// Add pattern matches if found
	if len(matches) > 0 {
		baseMsg += "\n\n📚 SIMILAR SHARP EDGES FOUND:\n"

		for i, match := range matches {
			baseMsg += fmt.Sprintf("\n**Match %d** (score: %d, matched on: %s):\n",
				i+1,
				match.Score,
				strings.Join(match.MatchedOn, ", "),
			)

			baseMsg += fmt.Sprintf("- **ID**: %s\n", match.Template.ID)
			baseMsg += fmt.Sprintf("- **Description**: %s\n", match.Template.Description)
			baseMsg += fmt.Sprintf("- **Suggested Solution**: %s\n", match.Template.Solution)
			baseMsg += fmt.Sprintf("- **Source**: %s\n", match.Template.Source)
		}

		baseMsg += "\n💡 Try the suggested solution from the highest-scored match."
	} else {
		baseMsg += "\n\nℹ️ No similar patterns found in sharp-edges.yaml. This may be a novel issue."
	}

	return &HookResponse{
		Decision: "block",
		Reason:   fmt.Sprintf("⚠️ SHARP EDGE DETECTED: %d consecutive failures on '%s' (%s)", failureCount, edge.File, edge.ErrorType),
		HookSpecificOutput: map[string]interface{}{
			"hookEventName":     "PostToolUse",
			"additionalContext": baseMsg,
		},
	}
}
```
