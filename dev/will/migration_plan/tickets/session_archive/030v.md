## Implementation Details

### .github/workflows/simulation.yml
```yaml
name: Simulation Tests

on:
  push:
    branches: [master]
  pull_request:
    branches: [master]
  schedule:
    - cron: '0 3 * * 0'  # Weekly Sunday 3am UTC
  workflow_dispatch:

env:
  GO_VERSION: '1.22'

jobs:
  simulation:
    name: Simulation Tests
    runs-on: ubuntu-latest
    timeout-minutes: 30

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}

      - name: Cache Go modules
        uses: actions/cache@v4
        with:
          path: |
            ~/.cache/go-build
            ~/go/pkg/mod
          key: ${{ runner.os }}-go-${{ hashFiles('**/go.sum') }}
          restore-keys: |
            ${{ runner.os }}-go-

      - name: Build CLIs
        run: make build-validate build-archive

      - name: Run Deterministic Tests
        id: deterministic
        run: |
          mkdir -p test/simulation/reports
          go run ./test/simulation/harness/cmd/harness \
            -mode=deterministic \
            -report=json \
            -output=test/simulation/reports || echo "det_failed=true" >> $GITHUB_ENV

      - name: Run Fuzz Tests
        id: fuzz
        run: |
          go run ./test/simulation/harness/cmd/harness \
            -mode=fuzz \
            -iterations=1000 \
            -seed=${{ github.run_number }} \
            -report=markdown \
            -output=test/simulation/reports || echo "fuzz_failed=true" >> $GITHUB_ENV

      - name: Upload Reports
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: simulation-reports-${{ github.run_number }}
          path: test/simulation/reports/
          retention-days: 30

      - name: Upload Crash Corpus
        uses: actions/upload-artifact@v4
        if: env.fuzz_failed == 'true'
        with:
          name: fuzz-crashes-${{ github.run_number }}
          path: test/simulation/tmp/fuzz/crashes/
          retention-days: 90
          if-no-files-found: ignore

      - name: Check Results
        run: |
          if [ "${{ env.det_failed }}" == "true" ]; then
            echo "::error::Deterministic tests failed"
            exit 1
          fi
          if [ "${{ env.fuzz_failed }}" == "true" ]; then
            echo "::error::Fuzz tests failed"
            exit 1
          fi
          echo "All simulation tests passed!"

  simulation-extended:
    name: Extended Fuzz (Weekly)
    runs-on: ubuntu-latest
    timeout-minutes: 120
    if: github.event_name == 'schedule' || github.event_name == 'workflow_dispatch'

    steps:
      - name: Checkout
        uses: actions/checkout@v4

      - name: Setup Go
        uses: actions/setup-go@v5
        with:
          go-version: ${{ env.GO_VERSION }}

      - name: Build CLIs
        run: make build-validate build-archive

      - name: Extended Fuzz Run
        run: |
          mkdir -p test/simulation/reports
          go run ./test/simulation/harness/cmd/harness \
            -mode=fuzz \
            -iterations=10000 \
            -timeout=90m \
            -seed=${{ github.run_number }} \
            -report=json \
            -output=test/simulation/reports

      - name: Upload Extended Reports
        uses: actions/upload-artifact@v4
        if: always()
        with:
          name: extended-fuzz-${{ github.run_number }}
          path: test/simulation/reports/
          retention-days: 90
```

### Makefile additions
```makefile
# ==============================================================================
# Simulation Testing
# ==============================================================================

.PHONY: test-simulation test-simulation-fuzz test-simulation-deterministic

# Run mixed simulation (deterministic + fuzz)
test-simulation: build-validate build-archive
	@echo "Running simulation tests (mixed mode)..."
	@mkdir -p test/simulation/reports
	go run ./test/simulation/harness/cmd/harness \
		-mode=mixed \
		-iterations=500 \
		-report=markdown \
		-output=test/simulation/reports
	@echo "Report: test/simulation/reports/"

# Run deterministic tests only
test-simulation-deterministic: build-validate build-archive
	@echo "Running deterministic simulation tests..."
	go run ./test/simulation/harness/cmd/harness \
		-mode=deterministic \
		-report=tap

# Run fuzz tests only
test-simulation-fuzz: build-validate build-archive
	@echo "Running fuzz simulation tests..."
	go run ./test/simulation/harness/cmd/harness \
		-mode=fuzz \
		-iterations=1000 \
		-verbose

# Replay a specific crash
# Usage: make replay-crash CRASH=path/to/crash.json
replay-crash: build-validate build-archive
	@if [ -z "$(CRASH)" ]; then \
		echo "Usage: make replay-crash CRASH=path/to/crash.json"; \
		exit 1; \
	fi
	go run ./test/simulation/harness/cmd/harness -replay=$(CRASH)

# Clean simulation artifacts
clean-simulation:
	rm -rf test/simulation/reports/*
	rm -rf test/simulation/tmp/*
```

## Workflow Triggers

| Event | Tests Run | Purpose |
|-------|-----------|---------|
| Push to master | Deterministic + Fuzz(1000) | Verify main branch |
| Pull request | Deterministic + Fuzz(1000) | Verify changes |
| Schedule (weekly) | Fuzz(10000) | Deep exploration |
| Manual dispatch | Fuzz(10000) | On-demand extended testing |

## Seed Strategy

Using `github.run_number` as seed provides:
- **Reproducibility**: Same run_number = same sequence
- **Variety**: Different builds explore different input space
- **Traceability**: Crash files include seed for exact replay

## Artifact Strategy

| Artifact | Retention | Purpose |
|----------|-----------|---------|
| Reports | 30 days | Review test results |
| Crashes | 90 days | Debug failures |
