### Sample Integration Test

```go
func TestWorkflow_ThreeFailures_Block(t *testing.T) {
    projectDir := setupTestProject(t)
    t.Setenv("GOGENT_PROJECT_DIR", projectDir)
    t.Setenv("GOGENT_MAX_FAILURES", "3")

    // Pre-populate 2 failures
    createTrackerEntries(t, projectDir, []FailureEntry{
        {Timestamp: time.Now().Unix() - 10, File: "test.py", ErrorType: "typeerror", Tool: "Bash"},
        {Timestamp: time.Now().Unix() - 5, File: "test.py", ErrorType: "typeerror", Tool: "Bash"},
    })

    // Simulate 3rd failure
    event := PostToolEvent{
        ToolName:     "Bash",
        ToolInput:    map[string]interface{}{"command": "python test.py"},
        ToolResponse: map[string]interface{}{"exit_code": 1.0, "output": "TypeError: foo"},
        CapturedAt:   time.Now().Unix(),
    }

    output := simulateFailure(t, projectDir, event)

    // Verify blocking response
    var resp map[string]interface{}
    require.NoError(t, json.Unmarshal([]byte(output), &resp))
    assert.Equal(t, "block", resp["decision"])
    assert.Contains(t, resp["reason"], "SHARP EDGE")

    // Verify pending-learnings.jsonl created
    learnings := readPendingLearnings(t, projectDir)
    require.Len(t, learnings, 1)
    assert.Equal(t, "test.py", learnings[0].File)
    assert.Equal(t, 3, learnings[0].ConsecutiveFailures)
}
```

### Sample Composite Key Test

```go
func TestCountRecentFailures_CompositeKey_SameFile(t *testing.T) {
    // Same file, 3 DIFFERENT error types → each count = 1
    tracker := setupTracker(t)

    // Log 3 different errors on same file
    tracker.LogFailure(FailureKey{"main.py", "TypeError", ""}, "Bash")
    tracker.LogFailure(FailureKey{"main.py", "ValueError", ""}, "Bash")
    tracker.LogFailure(FailureKey{"main.py", "SyntaxError", ""}, "Bash")

    // Each type should have count = 1
    count1, _ := tracker.CountRecentFailures(FailureKey{"main.py", "TypeError", ""})
    count2, _ := tracker.CountRecentFailures(FailureKey{"main.py", "ValueError", ""})
    count3, _ := tracker.CountRecentFailures(FailureKey{"main.py", "SyntaxError", ""})

    assert.Equal(t, 1, count1, "TypeError count")
    assert.Equal(t, 1, count2, "ValueError count")
    assert.Equal(t, 1, count3, "SyntaxError count")

    // This prevents false positives from triggering on "3 failures on main.py"
}
```

### Race Detection Test

```go
func TestStorage_ConcurrentWrites(t *testing.T) {
    tracker := setupTracker(t)
    var wg sync.WaitGroup

    // 10 goroutines writing 100 entries each
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func(goroutineID int) {
            defer wg.Done()
            for j := 0; j < 100; j++ {
                key := FailureKey{
                    FilePath:  fmt.Sprintf("file_%d.go", goroutineID),
                    ErrorType: "error",
                }
                tracker.LogFailure(key, "Bash")
            }
        }(i)
    }

    wg.Wait()

    // Verify all entries written (should be 1000 total)
    // Read file and count lines
    data, _ := os.ReadFile(tracker.ErrorLogPath)
    lines := strings.Split(strings.TrimSpace(string(data)), "\n")
    assert.Equal(t, 1000, len(lines))
}
```
