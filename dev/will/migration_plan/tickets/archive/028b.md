// formatPhaseMessage generates human-readable phase description
func formatPhaseMessage(phase string, dist map[string]int) string {
	switch phase {
	case "discovery":
		return "Exploration session - investigated codebase, gathered context"
	case "implementation":
		return "Implementation session - made code changes"
	case "debugging":
		return "Debugging session - tested and fixed issues"
	case "delegation":
		return "Coordination session - delegated work to agents"
	default:
		return "Mixed session - various activities"
	}
}

// getTopEditedFiles finds most frequently edited files from events
func getTopEditedFiles(events []ToolEvent) []string {
	fileCounts := make(map[string]int)

	for _, event := range events {
		if event.ToolName == "Edit" || event.ToolName == "Write" {
			if filePath, ok := event.ToolInput["file_path"].(string); ok && filePath != "" {
				fileCounts[filePath]++
			}
		}
	}

	if len(fileCounts) == 0 {
		return []string{}
	}

	// Convert to slice and sort by count (descending)
	type fileCount struct {
		path  string
		count int
	}

	files := make([]fileCount, 0, len(fileCounts))
	for path, count := range fileCounts {
		files = append(files, fileCount{path, count})
	}

	// Simple bubble sort for top 5 (small slice)
	for i := 0; i < len(files)-1; i++ {
		for j := 0; j < len(files)-i-1; j++ {
			if files[j].count < files[j+1].count {
				files[j], files[j+1] = files[j+1], files[j]
			}
		}
	}

	// Return top 5 (or fewer if less than 5 files)
	limit := 5
	if len(files) < limit {
		limit = len(files)
	}

	result := make([]string, limit)
	for i := 0; i < limit; i++ {
		result[i] = files[i].path
	}

	return result
}

// Enhanced GenerateHandoff with adaptive formatting
func GenerateHandoff(config *HandoffConfig, metrics *SessionMetrics, transcriptPath string) error {
	if config == nil {
		return fmt.Errorf("[handoff] Config nil. Cannot generate handoff without configuration. Provide valid HandoffConfig.")
	}

	if metrics == nil {
		return fmt.Errorf("[handoff] Metrics nil. Cannot generate handoff without session metrics. Provide valid SessionMetrics.")
	}

	// Ensure directory exists
	dir := filepath.Dir(config.HandoffPath)
	if err := os.MkdirAll(dir, 0755); err != nil {
		return fmt.Errorf("[handoff] Failed to create directory %s: %w. Check write permissions.", dir, err)
	}

	// Create handoff file
	f, err := os.Create(config.HandoffPath)
	if err != nil {
		return fmt.Errorf("[handoff] Failed to create file %s: %w. Check write permissions.", config.HandoffPath, err)
	}
	defer f.Close()

	// Write header
	timestamp := time.Now().Format("2006-01-02 15:04:05")
	fmt.Fprintf(f, "# Session Handoff - %s\n\n", timestamp)

	// Parse transcript if provided
	var events []ToolEvent
	var phases []SessionPhase
	var dist map[string]int
	var topFiles []string

	if transcriptPath != "" {
		events, _ = ParseTranscript(transcriptPath)
		if len(events) > 0 {
			phases = DetectPhases(events)
			dist = AnalyzeToolDistribution(events)
			topFiles = getTopEditedFiles(events)
		}
	}

	// Write session characterization (if transcript analyzed)
	if len(phases) > 0 {
		fmt.Fprintf(f, "## Session Characterization\n")
		phase := phases[0]
		fmt.Fprintf(f, "**Type**: %s\n", formatPhaseMessage(phase.Phase, dist))
		fmt.Fprintf(f, "**Phase**: %s\n", phase.Phase)

		if len(topFiles) > 0 {
			fmt.Fprintf(f, "**Focus**: %d edits across %d files\n", phase.ToolCount, len(topFiles))
			fmt.Fprintf(f, "**Top Edited Files**:\n")
			for _, file := range topFiles {
				fmt.Fprintf(f, "- %s\n", file)
			}
		}

		if len(dist) > 0 {
			fmt.Fprintf(f, "\n**Tool Distribution**:\n")
			for tool, count := range dist {
				fmt.Fprintf(f, "- %s: %d\n", tool, count)
			}
		}

		fmt.Fprintf(f, "\n")
	}

	// Write session metrics
	fmt.Fprintf(f, "## Session Metrics\n")
	fmt.Fprintf(f, "- **Tool Calls**: ~%d\n", metrics.ToolCallCount)
	fmt.Fprintf(f, "- **Errors Logged**: %d\n", metrics.ErrorCount)
	fmt.Fprintf(f, "- **Routing Violations**: %d\n", metrics.ViolationCount)
	fmt.Fprintf(f, "- **Session ID**: %s\n\n", metrics.SessionID)

	// ... (rest of handoff generation remains same as GOgent-028)

	return nil
}

	tests := []struct {
		phase    string
		expected string
	}{

		{"implementation", "Implementation session"},
