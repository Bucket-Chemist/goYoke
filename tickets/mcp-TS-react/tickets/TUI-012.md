# TUI-012: useClaudeQuery Hook

## Description

Implement the `useClaudeQuery` hook that wraps the SDK's `query()` function and connects it to the store for state updates.

## Acceptance Criteria

## Tasks

1. Create `src/hooks/useClaudeQuery.ts`:
   ```typescript
   import { query } from "@anthropic-ai/claude-agent-sdk";
   import { useCallback } from "react";
   import { useStore } from "../store/index.js";
   import { mcpServer } from "../mcp/server.js";

   export function useClaudeQuery() {
     const {
       addMessage,
       updateLastMessage,
       setStreaming,
       updateSession,
       addAgent,
       updateAgent
     } = useStore();

     const sendMessage = useCallback(async (content: string) => {
       // Add user message to store
       addMessage({
         role: "user",
         content: [{ type: "text", text: content }],
         partial: false
       });

       setStreaming(true);

       async function* messages() {
         yield {
           type: "user" as const,
           message: {
             role: "user" as const,
             content: [{ type: "text", text: content }]
           }
         };
       }

       try {
         for await (const event of query({
           prompt: messages(),
           options: {
             mcpServers: { "gofortress-interactive": mcpServer },
             allowedTools: [
               "mcp__gofortress-interactive__ask_user",
               "mcp__gofortress-interactive__confirm_action",
               "mcp__gofortress-interactive__request_input",
               "mcp__gofortress-interactive__select_option"
             ]
           }
         })) {
           handleEvent(event);
         }
       } catch (error) {
         handleError(error);
       } finally {
         setStreaming(false);
       }
     }, [addMessage, setStreaming, updateLastMessage]);

     return { sendMessage };
   }
   ```

   function handleEvent(event: QueryEvent): void {
     switch (event.type) {
       case "content_block_start":
         addMessage({ role: "assistant", content: [], partial: true });
         break;

       case "content_block_delta":
         // Update partial message content
         updateLastMessage(/* merged content */);
         break;

       case "content_block_stop":
         // Mark message as complete
         break;

       case "message_stop":
         // Finalize message
         break;

       case "usage":
         updateSession({
           tokenCount: {
             input: event.input_tokens,
             output: event.output_tokens
           }
         });
         break;

       // Handle agent/task events for tree
       case "task_start":
         addAgent({ /* agent data */ });
         break;

       case "task_complete":
         updateAgent(event.taskId, { status: "complete" });
         break;
     }
   }
