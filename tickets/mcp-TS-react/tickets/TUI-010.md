# TUI-010: Keyboard Handling & Timeout Logic

## Description

Implement comprehensive keyboard handling across the application and timeout logic for modals.

## Acceptance Criteria

- [ ] Global key bindings work (Tab, Escape, Ctrl+C, Ctrl+L)
- [ ] Panel-specific bindings only fire when focused
- [ ] Modal captures all input when active
- [ ] Modal timeout returns default/cancels appropriately
- [ ] Key bindings documented in help text
- [ ] No key binding conflicts

## Tasks

### Global Keyboard Hook

1. Create `src/hooks/useKeymap.ts`:
   ```typescript
   interface KeyBinding {
     key: string;
     ctrl?: boolean;
     meta?: boolean;
     shift?: boolean;
     action: () => void;
     description: string;
   }

   export function useKeymap(bindings: KeyBinding[]): void {
     useInput((input, key) => {
       for (const binding of bindings) {
         if (matchesBinding(input, key, binding)) {
           binding.action();
           return;
         }
       }
     });
   }
   ```

### Modal Timeout Logic

3. Update modal slice with timeout handling:
   ```typescript
   enqueue: (request) => {
     return new Promise((resolve, reject) => {
       const id = nanoid();
       let timeoutId: NodeJS.Timeout | undefined;

       const cleanup = () => {
         if (timeoutId) clearTimeout(timeoutId);
       };

       if (request.timeout) {
         timeoutId = setTimeout(() => {
           // Return default or cancel based on modal type
           if (request.type === "confirm") {
             resolve({ type: "confirm", confirmed: false, cancelled: true });
           } else {
             reject(new Error("Modal timeout"));
           }
           // Remove from queue
           set((state) => ({
             modalQueue: state.modalQueue.filter(m => m.id !== id)
           }));
         }, request.timeout);
       }

       const wrappedResolve = (response: ModalResponse) => {
         cleanup();
         resolve(response);
       };

       const wrappedReject = (error: Error) => {
         cleanup();
         reject(error);
       };

       set((state) => ({
         modalQueue: [...state.modalQueue, {
           ...request,
           id,
           resolve: wrappedResolve,
           reject: wrappedReject
         }]
       }));
     });
   }
   ```
