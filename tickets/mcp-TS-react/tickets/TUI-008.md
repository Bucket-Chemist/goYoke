### ClaudePanel

4. Create `src/components/ClaudePanel.tsx`:
   ```typescript
   export function ClaudePanel({ focused }: { focused: boolean }): JSX.Element {
     const { messages, streaming } = useStore();
     const { sendMessage } = useClaudeQuery();
     const [input, setInput] = useState("");

     const handleSubmit = () => {
       if (input.trim() && !streaming) {
         sendMessage(input);
         setInput("");
       }
     };

     return (
       <Box flexDirection="column" borderStyle="single" borderColor={focused ? "cyan" : "gray"}>
         {/* TODO: Import from theme.ts - colors.focused, colors.unfocused, etc. */}
         <Viewport items={messages} renderItem={renderMessage} />
         <Box borderStyle="single" borderColor="gray">
           <TextInput
             value={input}
             onChange={setInput}
             onSubmit={handleSubmit}
             placeholder={streaming ? "Waiting..." : "Type a message..."}
             disabled={streaming}
           />
           {streaming && <Spinner type="dots" />}
         </Box>
       </Box>
     );
   }
   ```
