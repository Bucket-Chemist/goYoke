### Update Build System

4. Update `Makefile`:
   ```makefile
   # New default: TypeScript TUI
   .PHONY: build
   build: build-hooks build-tui

   .PHONY: build-tui
   build-tui:
   	cd packages/tui && npm run build

   .PHONY: build-legacy
   build-legacy:
   	go build -o bin/gofortress-legacy ./deprecated/cmd/gofortress

   .PHONY: build-hooks
   build-hooks:
   	go build -o bin/gogent-load-context ./cmd/gogent-load-context
   	go build -o bin/gogent-validate ./cmd/gogent-validate
   	# ... other hooks
   ```

5. Update `bin/gofortress-tui.js` to handle `--legacy`:
   ```javascript
   if (process.argv.includes("--legacy")) {
     const { spawn } = require("child_process");
     spawn("./bin/gofortress-legacy", process.argv.slice(2), {
       stdio: "inherit"
     });
   } else {
     require("../dist/index.js");
   }
   ```
