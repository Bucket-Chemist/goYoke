# TUI-014: AgentTree Component

## Description

Implement the hierarchical agent tree view showing spawned agents, their status, and parent-child relationships.

## Acceptance Criteria

- [ ] Tree renders agent hierarchy correctly
- [ ] Root agent at top, children indented
- [ ] Status indicators visible (spawning/running/complete/error)
- [ ] Selected agent highlighted
- [ ] Tree scrolls for 20+ agents
- [ ] Performance acceptable with many agents
- [ ] **Uses theme constants from `src/config/theme.ts` (see TUI-003)**

## Tasks

1. Create `src/components/AgentTree.tsx`:
   ```typescript
   import { Box, Text } from "ink";
   import { useStore } from "../store/index.js";

   interface AgentTreeProps {
     focused: boolean;
   }

   export function AgentTree({ focused }: AgentTreeProps): JSX.Element {
     const { agents, selectedAgentId, rootAgentId } = useStore();

     const renderAgent = (agentId: string, depth: number): JSX.Element => {
       const agent = agents.get(agentId);
       if (!agent) return null;

       const isSelected = agentId === selectedAgentId;
       const children = getChildAgents(agents, agentId);

       return (
         <Box key={agentId} flexDirection="column">
           <Box paddingLeft={depth * 2}>
             {/* TODO: Import from theme.ts - colors.focused, colors.unfocused, etc. */}
             <Text
               inverse={isSelected}
               color={getStatusColor(agent.status)}
             >
               {getStatusIcon(agent.status)} {agent.model}
               {agent.description && `: ${agent.description}`}
             </Text>
           </Box>
           {children.map(childId => renderAgent(childId, depth + 1))}
         </Box>
       );
     };

     return (
       <Box
         flexDirection="column"
         borderStyle="single"
         {/* TODO: Import from theme.ts - colors.focused, colors.unfocused, etc. */}
         borderColor={focused ? "cyan" : "gray"}
       >
         <Text bold underline>Agents</Text>
         {rootAgentId && renderAgent(rootAgentId, 0)}
         {!rootAgentId && <Text dimColor>No agents yet</Text>}
       </Box>
     );
   }
   ```

2. Implement status helpers:
   ```typescript
   function getStatusIcon(status: Agent["status"]): string {
     switch (status) {
       case "spawning": return "◐";
       case "running": return "●";
       case "complete": return "✓";
       case "error": return "✗";
     }
   }

   // Import: import { colors } from "../config/theme.js";
   function getStatusColor(status: Agent["status"]): string {
     switch (status) {
       case "spawning": return colors.agentSpawning;
       case "running": return colors.agentRunning;
       case "complete": return colors.agentComplete;
       case "error": return colors.agentError;
     }
   }
   ```

3. Implement child agent lookup:
   ```typescript
   function getChildAgents(agents: Map<string, Agent>, parentId: string): string[] {
     return Array.from(agents.entries())
       .filter(([_, agent]) => agent.parentId === parentId)
       .map(([id]) => id);
   }
   ```
