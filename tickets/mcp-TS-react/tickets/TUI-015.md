# TUI-015: AgentDetail & Tree Navigation

## Description

Implement the agent detail panel showing selected agent information and keyboard navigation for the agent tree.

## Acceptance Criteria

- [ ] AgentDetail shows selected agent info
- [ ] Displays: model, tier, status, duration, tokens
- [ ] Up/Down arrows navigate tree
- [ ] Enter expands/collapses (if applicable)
- [ ] Navigation wraps at boundaries
- [ ] Empty state when no agent selected

## Tasks

### AgentDetail Component

1. Create `src/components/AgentDetail.tsx`:
   ```typescript
   import { Box, Text } from "ink";
   import { useStore } from "../store/index.js";

   export function AgentDetail(): JSX.Element {
     const { agents, selectedAgentId } = useStore();

     if (!selectedAgentId) {
       return (
         <Box borderStyle="single" padding={1}>
           <Text dimColor>Select an agent to view details</Text>
         </Box>
       );
     }

     const agent = agents.get(selectedAgentId);
     if (!agent) return null;

     const duration = agent.endTime
       ? formatDuration(agent.endTime - agent.startTime)
       : "Running...";

     return (
       <Box flexDirection="column" borderStyle="single" padding={1}>
         <Text bold>{agent.model}</Text>
         <Text>Tier: {agent.tier}</Text>
         <Text>Status: <Text color={getStatusColor(agent.status)}>{agent.status}</Text></Text>
         <Text>Duration: {duration}</Text>
         {agent.tokenUsage && (
           <>
             <Text>Input tokens: {agent.tokenUsage.input}</Text>
             <Text>Output tokens: {agent.tokenUsage.output}</Text>
           </>
         )}
         {agent.description && (
           <Box marginTop={1}>
             <Text wrap="wrap">{agent.description}</Text>
           </Box>
         )}
       </Box>
     );
   }
   ```

### Tree Navigation Hook

2. Create `src/hooks/useAgentTree.ts`:
   ```typescript
   import { useCallback } from "react";
   import { useStore } from "../store/index.js";

   export function useAgentTree() {
     const { agents, selectedAgentId, selectAgent, rootAgentId } = useStore();

     // Flatten tree to navigable list
     const flattenedIds = useMemo(() => {
       const result: string[] = [];
       const traverse = (id: string) => {
         result.push(id);
         getChildAgents(agents, id).forEach(traverse);
       };
       if (rootAgentId) traverse(rootAgentId);
       return result;
     }, [agents, rootAgentId]);

     const selectPrevious = useCallback(() => {
       const currentIndex = flattenedIds.indexOf(selectedAgentId ?? "");
       const newIndex = Math.max(0, currentIndex - 1);
       selectAgent(flattenedIds[newIndex] ?? null);
     }, [flattenedIds, selectedAgentId, selectAgent]);

     const selectNext = useCallback(() => {
       const currentIndex = flattenedIds.indexOf(selectedAgentId ?? "");
       const newIndex = Math.min(flattenedIds.length - 1, currentIndex + 1);
       selectAgent(flattenedIds[newIndex] ?? null);
     }, [flattenedIds, selectedAgentId, selectAgent]);

     return { selectPrevious, selectNext, flattenedIds };
   }
   ```

3. Integrate navigation with Layout:
   ```typescript
   // In Layout or AgentTree when focused
   useInput((input, key) => {
     if (!focused) return;

     if (key.upArrow) {
       selectPrevious();
     }
     if (key.downArrow) {
       selectNext();
     }
   });
   ```

## Utility Functions

```typescript
function formatDuration(ms: number): string {
  if (ms < 1000) return `${ms}ms`;
  if (ms < 60000) return `${(ms / 1000).toFixed(1)}s`;
  return `${Math.floor(ms / 60000)}m ${((ms % 60000) / 1000).toFixed(0)}s`;
}
```
