### Benchmark Script

1. Create `scripts/benchmark.ts`:
   ```typescript
   import { execSync, spawn } from "child_process";
   import { performance } from "perf_hooks";

   interface BenchmarkResult {
     measured_at: string;
     cold_start_ms: number;
     memory_idle_mb: number;
     memory_active_mb: number;
     input_latency_ms: number;
   }

   async function measureColdStart(): Promise<number> {
     const runs: number[] = [];

     for (let i = 0; i < 5; i++) {
       const start = performance.now();
       execSync("node dist/index.js --version", { stdio: "ignore" });
       runs.push(performance.now() - start);
     }

     // Return median
     runs.sort((a, b) => a - b);
     return runs[Math.floor(runs.length / 2)];
   }

   async function measureMemory(): Promise<{ idle: number; active: number }> {
     // Spawn TUI, measure memory after startup
     const child = spawn("node", ["dist/index.js"], {
       stdio: ["pipe", "pipe", "pipe"]
     });

     await new Promise(r => setTimeout(r, 2000)); // Wait for startup

     const pid = child.pid;
     const rss = parseInt(
       execSync(`ps -o rss= -p ${pid}`).toString().trim()
     );
     const idleMb = rss / 1024;

     // Simulate activity and measure again
     // ...

     child.kill();
     return { idle: idleMb, active: idleMb * 1.5 }; // Placeholder
   }

   // Run all benchmarks
   async function main(): Promise<void> {
     console.log("Running benchmarks...\n");

     const coldStart = await measureColdStart();
     const memory = await measureMemory();

     const result: BenchmarkResult = {
       measured_at: new Date().toISOString(),
       cold_start_ms: coldStart,
       memory_idle_mb: memory.idle,
       memory_active_mb: memory.active,
       input_latency_ms: 0 // TODO: measure
     };

     console.log("Results:", JSON.stringify(result, null, 2));

     // Compare with Go baseline
     const baseline = JSON.parse(
       readFileSync(".claude/tmp/go-baseline.json", "utf-8")
     );

     console.log("\nComparison:");
     console.log(`Cold start: ${result.cold_start_ms}ms vs ${baseline.cold_start_ms}ms (Go)`);
     console.log(`Memory idle: ${result.memory_idle_mb}MB vs ${baseline.memory_idle_mb}MB (Go)`);
   }

   main();
   ```
