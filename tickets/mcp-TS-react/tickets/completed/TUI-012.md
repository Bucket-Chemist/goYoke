# TUI-012: useClaudeQuery Hook

## Description

Implement the `useClaudeQuery` hook that wraps the SDK's `query()` function and connects it to the store for state updates.

**Note:** Event types verified in TUI-001. SDK uses `system`, `assistant`, `result` events (async iterable), not granular content block events. See `.claude/tmp/sdk-verification.json` for details.

1. Create `src/hooks/useClaudeQuery.ts`:
   ```typescript
   import { query } from "@anthropic-ai/claude-agent-sdk";
   import { useCallback } from "react";
   import { useStore } from "../store/index.js";
   import { mcpServer } from "../mcp/server.js";

   export function useClaudeQuery() {
     const {
       addMessage,
       updateLastMessage,
       setStreaming,
       updateSession,
       addAgent,
       updateAgent
     } = useStore();

     const sendMessage = useCallback(async (content: string) => {
       // Add user message to store
       addMessage({
         role: "user",
         content: [{ type: "text", text: content }],
         partial: false
       });

       setStreaming(true);

       async function* messages() {
         yield {
           type: "user" as const,
           message: {
             role: "user" as const,
             content: [{ type: "text", text: content }]
           }
         };
       }

       try {
         for await (const event of query({
           prompt: messages(),
           options: {
             mcpServers: { "gofortress-interactive": mcpServer },
             allowedTools: [
               "mcp__gofortress-interactive__ask_user",
               "mcp__gofortress-interactive__confirm_action",
               "mcp__gofortress-interactive__request_input",
               "mcp__gofortress-interactive__select_option"
             ]
           }
         })) {
           handleEvent(event);
         }
       } catch (error) {
         handleError(error);
       } finally {
         setStreaming(false);
       }
     }, [addMessage, setStreaming, updateLastMessage]);

     return { sendMessage };
   }
   ```

   function handleEvent(event: QueryEvent): void {
     switch (event.type) {
       case "system":
         // System initialization event
         // Contains: cwd, session_id, tools, mcp_servers, model, agents, skills
         updateSession({
           sessionId: event.session_id,
           model: event.model,
           tools: event.tools,
           mcpServers: event.mcp_servers
         });

         // Initialize agents from system event if present
         if (event.agents) {
           event.agents.forEach(agent => addAgent(agent));
         }
         break;

       case "assistant":
         // Assistant message event (can be partial during streaming)
         // Contains: message, parent_tool_use_id, session_id
         if (event.message) {
           // Check if this is updating an existing partial message
           // or starting a new one
           const isUpdate = /* check if message exists */;

           if (isUpdate) {
             updateLastMessage({
               content: event.message.content,
               partial: true
             });
           } else {
             addMessage({
               role: "assistant",
               content: event.message.content,
               partial: true,
               parentToolUseId: event.parent_tool_use_id
             });
           }
         }
         break;

       case "result":
         // Final result event with usage and completion info
         // Contains: is_error, duration_ms, num_turns, result, total_cost_usd, usage

         // Mark last message as complete
         updateLastMessage({ partial: false });

         // Update session with usage stats
         updateSession({
           tokenCount: {
             input: event.usage.input_tokens,
             output: event.usage.output_tokens
           },
           totalCost: event.total_cost_usd,
           numTurns: event.num_turns,
           durationMs: event.duration_ms
         });

         // Handle error state if present
         if (event.is_error) {
           handleError(event.result);
         }
         break;
     }
   }
