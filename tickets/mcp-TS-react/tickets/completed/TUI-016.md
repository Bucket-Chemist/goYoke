### Session File Operations

2. Create `src/hooks/useSession.ts`:
   ```typescript
   import { readFile, writeFile, readdir } from "fs/promises";
   import { join } from "path";

   const SESSION_DIR = join(process.env.HOME!, ".claude", "sessions");

   interface SessionFile {
     id: string;
     name?: string;
     created_at: string;  // ISO 8601
     last_used: string;   // ISO 8601
     cost: number;
     tool_calls: number;
   }

   export async function listSessions(): Promise<SessionFile[]> {
     const files = await readdir(SESSION_DIR);
     const sessions: SessionFile[] = [];

     for (const file of files.filter(f => f.endsWith(".json"))) {
       try {
         const content = await readFile(join(SESSION_DIR, file), "utf-8");
         sessions.push(JSON.parse(content));
       } catch {
         // Skip invalid files
       }
     }

     return sessions.sort((a, b) =>
       new Date(b.last_used).getTime() - new Date(a.last_used).getTime()
     );
   }

   export async function loadSession(id: string): Promise<SessionFile | null> {
     try {
       const content = await readFile(
         join(SESSION_DIR, `${id}.json`),
         "utf-8"
       );
       return JSON.parse(content);
     } catch {
       return null;
     }
   }

   export async function saveSession(session: SessionFile): Promise<void> {
     await writeFile(
       join(SESSION_DIR, `${session.id}.json`),
       JSON.stringify(session, null, 2)
     );
   }
   ```

### List Sessions Display

3. Create `src/commands/list.ts`:
   ```typescript
   export async function listSessionsCommand(): Promise<void> {
     const sessions = await listSessions();

     if (sessions.length === 0) {
       console.log("No sessions found.");
       return;
     }

     console.log("Available sessions:\n");
     for (const session of sessions) {
       const date = new Date(session.last_used).toLocaleDateString();
       const cost = `$${session.cost.toFixed(4)}`;
       console.log(`  ${session.id}  ${date}  ${cost}  ${session.name ?? ""}`);
     }
   }
   ```
