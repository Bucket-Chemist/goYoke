### Test Pattern: Table-Driven with Subtests

All tests follow Go's table-driven pattern from go.md conventions:

```
func TestFunctionName(t *testing.T) {
    tests := []struct {
        name     string
        input    InputType
        expected OutputType
        wantErr  bool
    }{
        {
            name:     "descriptive test case name",
            input:    /* ... */,
            expected: /* ... */,
        },
        // ... more cases
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            t.Parallel()  // Enable parallel execution

            result, err := FunctionName(tc.input)

            if tc.wantErr {
                require.Error(t, err)
                return
            }
            require.NoError(t, err)
            assert.Equal(t, tc.expected, result)
        })
    }
}
```

func TestLoadConfig(t *testing.T) {
    tests := []struct {
        name     string
        setupDir func(t *testing.T) string  // Returns temp dir path
        wantErr  bool
        validate func(t *testing.T, cfg *TeamConfig)
    }{
        {
            name: "valid braintrust config",
            setupDir: func(t *testing.T) string {
                dir := t.TempDir()
                config := `{
                    "team_name": "test-braintrust",
                    "workflow_type": "braintrust",
                    "project_root": "/absolute/path",
                    "budget_max_usd": 5.0,
                    "budget_remaining_usd": 5.0,
                    "waves": [{"wave_number": 1, "members": []}]
                }`
                os.WriteFile(filepath.Join(dir, "config.json"), []byte(config), 0644)
                return dir
            },
            validate: func(t *testing.T, cfg *TeamConfig) {
                assert.Equal(t, "test-braintrust", cfg.TeamName)
                assert.Equal(t, 5.0, cfg.BudgetMaxUSD)
                assert.Equal(t, "/absolute/path", cfg.ProjectRoot)
            },
        },
        {
            name: "missing project_root",
            setupDir: func(t *testing.T) string {
                dir := t.TempDir()
                config := `{"team_name": "test", "waves": []}`
                os.WriteFile(filepath.Join(dir, "config.json"), []byte(config), 0644)
                return dir
            },
            wantErr: true,
        },
        {
            name: "relative project_root",
            setupDir: func(t *testing.T) string {
                dir := t.TempDir()
                config := `{"project_root": "relative/path", "waves": []}`
                os.WriteFile(filepath.Join(dir, "config.json"), []byte(config), 0644)
                return dir
            },
            wantErr: true,
        },
        {
            name: "malformed JSON",
            setupDir: func(t *testing.T) string {
                dir := t.TempDir()
                os.WriteFile(filepath.Join(dir, "config.json"), []byte("{invalid json}"), 0644)
                return dir
            },
            wantErr: true,
        },
        {
            name: "missing file",
            setupDir: func(t *testing.T) string {
                return t.TempDir()  // Empty directory
            },
            wantErr: true,
        },
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            t.Parallel()

            dir := tc.setupDir(t)
            cfg, err := loadConfig(dir)

            if tc.wantErr {
                require.Error(t, err)
                return
            }

            require.NoError(t, err)
            if tc.validate != nil {
                tc.validate(t, cfg)
            }
        })
    }
}

func TestBuildPromptEnvelope(t *testing.T) {
    tests := []struct {
        name      string
        stdinFile string       // JSON content
        member    *TeamMember
        wantErr   bool
        validate  func(t *testing.T, envelope string)
    }{
        {
            name: "absolute paths resolved",
            stdinFile: `{
                "$schema": "stdin-einstein.json",
                "task": "Analyze problem",
                "paths": {
                    "project_root": "/abs/path",
                    "files": ["main.go", "test.go"]
                }
            }`,
            member: &TeamMember{
                Agent:      "einstein",
                StdinFile:  "stdin_einstein.json",
                StdoutFile: "stdout_einstein.json",
            },
            validate: func(t *testing.T, envelope string) {
                assert.Contains(t, envelope, "AGENT: einstein")
                assert.Contains(t, envelope, "Analyze problem")
                // Verify relative paths converted to absolute
                assert.Contains(t, envelope, "/abs/path/main.go")
            },
        },
        {
            name: "missing stdin file",
            member: &TeamMember{
                StdinFile: "nonexistent.json",
            },
            wantErr: true,
        },
        {
            name: "malformed stdin JSON",
            stdinFile: "{invalid",
            member: &TeamMember{
                StdinFile: "stdin_bad.json",
            },
            wantErr: true,
        },
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            t.Parallel()

            dir := t.TempDir()
            if tc.stdinFile != "" {
                path := filepath.Join(dir, tc.member.StdinFile)
                os.WriteFile(path, []byte(tc.stdinFile), 0644)
            }

            envelope, err := buildPromptEnvelope(dir, tc.member)

            if tc.wantErr {
                require.Error(t, err)
                return
            }

            require.NoError(t, err)
            if tc.validate != nil {
                tc.validate(t, envelope)
            }
        })
    }
}

                "content": {}

            errMsg:  "parse",

        {
            name:    "invalid JSON",
            content: "{not valid json}",
            wantErr: true,
            errMsg:  "parse",
        },
        {
            name: "status failed but valid envelope",
            content: `{
                "$schema": "stdout-einstein.json",
                "status": "failed",
                "content": {}
            }`,
            // Not an error - validation only checks envelope structure
        },

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            t.Parallel()

func TestWaveOrdering(t *testing.T) {
    tests := []struct {
        name     string
        waves    []Wave
        validate func(t *testing.T, execOrder []string)
    }{
        {
            name: "wave 2 waits for wave 1",
            waves: []Wave{
                {WaveNumber: 1, Members: []TeamMember{
                    {Name: "member1", Agent: "test-agent"},
                }},
                {WaveNumber: 2, Members: []TeamMember{
                    {Name: "member2", Agent: "test-agent"},
                }},
            },
            validate: func(t *testing.T, execOrder []string) {
                // Verify member1 completed before member2 started
                idx1 := findIndex(execOrder, "member1_completed")
                idx2 := findIndex(execOrder, "member2_started")
                assert.Less(t, idx1, idx2, "Wave 1 must complete before Wave 2 starts")
            },
        },
        {
            name: "members within wave execute in parallel",
            waves: []Wave{
                {WaveNumber: 1, Members: []TeamMember{
                    {Name: "member1", Agent: "test-agent"},
                    {Name: "member2", Agent: "test-agent"},
                    {Name: "member3", Agent: "test-agent"},
                }},
            },
            validate: func(t *testing.T, execOrder []string) {
                // All started events should come before all completed events
                // (proving concurrent execution)
                starts := filterPrefix(execOrder, "_started")
                completes := filterPrefix(execOrder, "_completed")
                assert.Len(t, starts, 3)
                assert.Len(t, completes, 3)
            },
        },
        {
            name: "single wave works",
            waves: []Wave{
                {WaveNumber: 1, Members: []TeamMember{
                    {Name: "solo", Agent: "test-agent"},
                }},
            },
            validate: func(t *testing.T, execOrder []string) {
                assert.Contains(t, execOrder, "solo_started")
                assert.Contains(t, execOrder, "solo_completed")
            },
        },
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            // Use mock spawner that tracks execution order
            execOrder := []string{}
            var mu sync.Mutex

            mockSpawn := func(member *TeamMember) error {
                mu.Lock()
                execOrder = append(execOrder, member.Name+"_started")
                mu.Unlock()

                // Simulate work
                time.Sleep(10 * time.Millisecond)

                mu.Lock()
                execOrder = append(execOrder, member.Name+"_completed")
                mu.Unlock()

                return nil
            }

            // Run wave scheduler with mock
            tr := &TeamRunner{config: &TeamConfig{Waves: tc.waves}}
            err := tr.runWavesWithMock(mockSpawn)

            require.NoError(t, err)
            tc.validate(t, execOrder)
        })
    }
}

```
func TestBudgetGate(t *testing.T) {
    tests := []struct {
        name              string
        budgetRemaining   float64
        memberCosts       []float64  // Cost of each member in sequence
        expectBlocked     int         // Index of member that should be blocked
    }{
        {
            name:            "sufficient budget for all",
            budgetRemaining: 1.0,
            memberCosts:     []float64{0.2, 0.3, 0.4},
            expectBlocked:   -1,  // None blocked
        },
        {
            name:            "budget exhausted after 2 members",
            budgetRemaining: 0.5,
            memberCosts:     []float64{0.2, 0.25, 0.3},
            expectBlocked:   2,  // Third member blocked
        },
        {
            name:            "zero budget blocks all",
            budgetRemaining: 0.0,
            memberCosts:     []float64{0.1, 0.1},
            expectBlocked:   0,  // First member blocked
        },
    }

    for _, tc := range tests {
        t.Run(tc.name, func(t *testing.T) {
            config := &TeamConfig{
                BudgetRemainingUSD: tc.budgetRemaining,
                Waves: []Wave{{
                    WaveNumber: 1,
                    Members:    make([]TeamMember, len(tc.memberCosts)),
                }},
            }

            for i := range config.Waves[0].Members {
                config.Waves[0].Members[i].Name = fmt.Sprintf("member%d", i)
            }

            tr := &TeamRunner{config: config}

            // Mock spawner that deducts cost
            spawnCount := 0
            mockSpawn := func(member *TeamMember) error {
                idx := spawnCount
                spawnCount++

                if idx >= len(tc.memberCosts) {
                    return fmt.Errorf("unexpected spawn call")
                }

                cost := tc.memberCosts[idx]
                tr.configMu.Lock()
                tr.config.BudgetRemainingUSD -= cost
                tr.configMu.Unlock()

                return nil
            }

            err := tr.runWavesWithMock(mockSpawn)

            if tc.expectBlocked == -1 {
                require.NoError(t, err)
                assert.Equal(t, len(tc.memberCosts), spawnCount)
            } else {
                // Verify spawn stopped at expected index
                assert.Equal(t, tc.expectBlocked, spawnCount)
            }
        })
    }
}
```

```
func TestTeamRunnerUpdateMemberConcurrent(t *testing.T) {
    config := &TeamConfig{
        Waves: []Wave{{
            WaveNumber: 1,
            Members: []TeamMember{
                {Name: "member1", Status: "pending"},
                {Name: "member2", Status: "pending"},
                {Name: "member3", Status: "pending"},
                {Name: "member4", Status: "pending"},
            },
        }},
    }

    dir := t.TempDir()
    tr := &TeamRunner{
        teamDir: dir,
        config:  config,
    }

    // 4 goroutines updating 4 different members concurrently
    var wg sync.WaitGroup
    for i := 0; i < 4; i++ {
        wg.Add(1)
        go func(idx int) {
            defer wg.Done()

            for attempt := 0; attempt < 10; attempt++ {
                err := tr.updateMember(0, idx, func(m *TeamMember) {
                    m.RetryCount = attempt
                    m.Status = "running"
                })
                require.NoError(t, err)

                time.Sleep(1 * time.Millisecond)

                err = tr.updateMember(0, idx, func(m *TeamMember) {
                    m.Status = "completed"
                })
                require.NoError(t, err)
            }
        }(i)
    }

    wg.Wait()

    // Verify all members ended in "completed" state
    for i, member := range tr.config.Waves[0].Members {
        assert.Equal(t, "completed", member.Status, "member%d status", i)
        assert.Equal(t, 9, member.RetryCount, "member%d retry count", i)
    }

    // Verify config.json was written atomically (no corruption)
    data, err := os.ReadFile(filepath.Join(dir, "config.json"))
    require.NoError(t, err)

    var reloaded TeamConfig
    require.NoError(t, json.Unmarshal(data, &reloaded))
}
```

## Code Guidelines

From go.md conventions:

1. **Table-driven tests**: Every test function uses table pattern
2. **Subtests with t.Run**: Enables parallel execution and better failure reporting
3. **t.Parallel()**: Mark independent tests as parallel for faster runs
4. **require vs assert**: Use `require` for critical checks (stops test), `assert` for informational
5. **t.TempDir()**: Automatic cleanup of test directories
6. **testify packages**: `github.com/stretchr/testify/assert` and `require`
