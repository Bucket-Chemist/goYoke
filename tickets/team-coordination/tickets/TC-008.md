**TeamRunner** (orchestrates execution with mutex-protected config):
```
type TeamRunner struct {
    teamDir   string
    config    *TeamConfig
    configMu  sync.Mutex

    childPIDs  map[int]struct{}  // Track spawned PIDs (for signal cascade)
    childrenMu sync.Mutex

    ctx       context.Context
    cancel    context.CancelFunc
}

// updateMember provides mutex-protected member updates (from TC-002)
func (tr *TeamRunner) updateMember(wavNum, memberIdx int, fn func(*TeamMember)) error {
    tr.configMu.Lock()
    defer tr.configMu.Unlock()

    if waveNum >= len(tr.config.Waves) || memberIdx >= len(tr.config.Waves[waveNum].Members) {
        return fmt.Errorf("invalid wave/member index")
    }

    member := &tr.config.Waves[waveNum].Members[memberIdx]
    fn(member)  // Apply mutation

    // Atomic write to config.json
    return tr.writeConfigAtomic()
}

// writeConfigAtomic implements write-tmp-then-rename pattern
func (tr *TeamRunner) writeConfigAtomic() error {
    tmpPath := filepath.Join(tr.teamDir, "config.json.tmp")
    finalPath := filepath.Join(tr.teamDir, "config.json")

    data, err := json.MarshalIndent(tr.config, "", "  ")
    if err != nil {
        return fmt.Errorf("marshal config: %w", err)
    }

    if err := os.WriteFile(tmpPath, data, 0644); err != nil {
        return fmt.Errorf("write tmp config: %w", err)
    }

    if err := os.Rename(tmpPath, finalPath); err != nil {
        return fmt.Errorf("rename config: %w", err)
    }

    return nil
}
```

```
func loadConfig(teamDir string) (*TeamConfig, error) {
    configPath := filepath.Join(teamDir, "config.json")

    // Check for abandoned .tmp file (from previous crash)
    tmpPath := configPath + ".tmp"
    if _, err := os.Stat(tmpPath); err == nil {
        log.Printf("[WARN] Found abandoned %s — previous process may have crashed during write. Removing.", tmpPath)
        os.Remove(tmpPath)
    }

    data, err := os.ReadFile(configPath)
    if err != nil {
        return nil, fmt.Errorf("read config.json: %w", err)
    }

    var config TeamConfig
    if err := json.Unmarshal(data, &config); err != nil {
        return nil, fmt.Errorf("parse config.json: %w", err)
    }

    // Validate required fields
    if config.ProjectRoot == "" {
        return nil, fmt.Errorf("project_root is required (from TC-006)")
    }
    if !filepath.IsAbs(config.ProjectRoot) {
        return nil, fmt.Errorf("project_root must be absolute path")
    }

    return &config, nil
}

func validateAgents(config *TeamConfig) error {
    index, err := loadAgentsIndex()
    if err != nil { return fmt.Errorf("load agents-index.json: %w", err) }
    for _, wave := range config.Waves {
        for _, member := range wave.Members {
            if _, ok := index[member.Agent]; !ok {
                return fmt.Errorf("unknown agent %q for member %q — check agents-index.json", member.Agent, member.Name)
            }
        }
    }
    return nil
}

// tryReserveBudget atomically checks AND reserves budget for an agent spawn.
// Returns false if insufficient budget remains.
func (tr *TeamRunner) tryReserveBudget(estimatedCost float64) bool {
    tr.configMu.Lock()
    defer tr.configMu.Unlock()
    if tr.config.BudgetRemainingUSD < estimatedCost {
        return false
    }
    tr.config.BudgetRemainingUSD -= estimatedCost
    tr.writeConfigAtomic()
    return true
}

// reconcileCost returns the budget reservation and deducts actual cost.
func (tr *TeamRunner) reconcileCost(estimatedCost, actualCost float64) {
    tr.configMu.Lock()
    defer tr.configMu.Unlock()
    tr.config.BudgetRemainingUSD += estimatedCost
    tr.config.BudgetRemainingUSD -= actualCost
    tr.writeConfigAtomic()
}

func estimateCost(agentID string) float64 {
    defaults := map[string]float64{
        "opus": 1.50, "sonnet": 0.30, "haiku": 0.05,
    }
    return defaults["opus"] // Conservative fallback
}
```

#### 2. Wave Scheduler (`wave.go`)

```
func (tr *TeamRunner) runWaves() error {
    for waveIdx, wave := range tr.config.Waves {
        log.Printf("[Wave %d] Starting with %d members", wave.WaveNumber, len(wave.Members))

        // Check budget before wave
        if tr.config.BudgetRemainingUSD <= 0 {
            return fmt.Errorf("budget exhausted before wave %d", wave.WaveNumber)
        }

        // Execute wave members in parallel
        var wg sync.WaitGroup
        for memberIdx := range wave.Members {
            estimated := estimateCost(wave.Members[memberIdx].Agent)
            if !tr.tryReserveBudget(estimated) {
                log.Printf("[Budget Gate] Cannot reserve $%.2f for %s", estimated, wave.Members[memberIdx].Name)
                break
            }
            wg.Add(1)
            go tr.spawnAndWait(waveIdx, memberIdx, &wg, estimated)
        }

        // Wait for all wave members to complete
        wg.Wait()

        // Execute inter-wave script if defined (TC-010)
        if wave.OnCompleteScript != "" {
            if err := tr.runInterWaveScript(wave.OnCompleteScript); err != nil {
                return fmt.Errorf("wave %d inter-wave script: %w", wave.WaveNumber, err)
            }
        }

        log.Printf("[Wave %d] Completed", wave.WaveNumber)
    }

    return nil
}
```

#### 3. Process Spawner (`spawn.go`)

> **Review Finding (Item 13)**: All budget mutations go through `tryReserveBudget()` and `reconcileCost()`. Direct `tr.configMu.Lock()` for budget changes is prohibited. Only `tryReserveBudget()` and `reconcileCost()` should touch `BudgetRemainingUSD`.

```
// spawnAndWait spawns a Claude CLI process and waits for completion with retries.
//
// CONTRACT: The caller (runWaves) must call wg.Add(1) exactly once before
// spawning this goroutine. This function calls wg.Done() exactly once via
// defer, regardless of retry count. Do NOT call this function without
// wg.Add(1) — the WaitGroup will go negative and panic.
func (tr *TeamRunner) spawnAndWait(waveIdx, memberIdx int, wg *sync.WaitGroup, estimatedCost float64) {
    defer wg.Done()  // Single defer - matches single wg.Add(1) (from TC-003)

    member := &tr.config.Waves[waveIdx].Members[memberIdx]

    for attempt := 0; attempt <= member.MaxRetries; attempt++ {
        // Update status to running
        tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
            m.Status = "running"
            m.RetryCount = attempt
            m.StartedAt = time.Now().Unix()
        })

        // Build prompt envelope from stdin file (TC-007)
        envelope, err := buildPromptEnvelope(tr.teamDir, member)
        if err != nil {
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.ErrorMessage = fmt.Sprintf("build envelope: %v", err)
            })
            continue  // Retry
        }

        // Load agent config for CLI flags (TC-014)
        agentConfig, err := loadAgentConfig(member.Agent)
        if err != nil {
            log.Printf("[WARN] Failed to load agent config for %s: %v", member.Agent, err)
            // Continue with default tools
            agentConfig = &AgentConfig{AllowedTools: []string{"Read", "Write", "Edit", "Bash", "Glob", "Grep"}}
        }

        // Build CLI args (TC-001)
        args := buildCLIArgs(agentConfig.AllowedTools)

        // Set per-agent timeout
        timeout := time.Duration(member.TimeoutMS) * time.Millisecond
        if timeout == 0 {
            timeout = 10 * time.Minute
        }
        agentCtx, agentCancel := context.WithTimeout(tr.ctx, timeout)

        // Spawn claude CLI process
        cmd := exec.CommandContext(agentCtx, "claude", args...)
        cmd.Dir = tr.config.ProjectRoot  // From TC-006
        cmd.Stdin = strings.NewReader(envelope)
        cmd.SysProcAttr = &syscall.SysProcAttr{
            Setsid: true,  // Process group isolation (TC-004)
        }

        // Capture stdout for cost extraction
        var stdoutBuf bytes.Buffer
        cmd.Stdout = &stdoutBuf

        // Start process
        if err := cmd.Start(); err != nil {
            agentCancel()
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.ErrorMessage = fmt.Sprintf("spawn attempt %d: %v", attempt, err)
            })
            continue  // Retry
        }

        // Register PID for signal cascade (atomic with Start())
        pid := cmd.Process.Pid
        tr.registerChild(pid)
        defer tr.unregisterChild(pid)

        tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
            m.ProcessPID = pid
        })

        // Wait for completion
        err = cmd.Wait()
        agentCancel()

        // Check for timeout
        if agentCtx.Err() == context.DeadlineExceeded {
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.ErrorMessage = fmt.Sprintf("timeout after %v", timeout)
            })
            continue
        }

        if err != nil {
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.ErrorMessage = fmt.Sprintf("wait attempt %d: %v", attempt, err)
            })
            continue  // Retry
        }

        // Extract cost from CLI output (TC-005)
        actualCost, costErr := extractCostFromCLIOutput(stdoutBuf.Bytes())
        if costErr != nil {
            if costErr.Level == "error" {
                log.Printf("[ERROR] cost: %s for %s\n  → %s", costErr.Message, member.Name, costErr.RawOutput[:min(len(costErr.RawOutput), 200)])
                tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                    m.CostStatus = "error"
                })
            } else {
                log.Printf("[WARN] cost: %s for %s", costErr.Message, member.Name)
                tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                    m.CostStatus = "unknown"
                })
            }
            actualCost = 0  // Continue without cost
        } else {
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.CostStatus = "ok"
            })
        }

        // Reconcile budget (return reservation, deduct actual)
        tr.reconcileCost(estimatedCost, actualCost)

        // Validate stdout envelope
        if err := validateStdout(member.StdoutFile); err != nil {
            tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
                m.ErrorMessage = fmt.Sprintf("validate stdout: %v", err)
            })
            continue  // Retry
        }

        // Success path
        tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
            m.Status = "completed"
            m.CompletedAt = time.Now().Unix()
            m.CostUSD = actualCost
        })
        return
    }

    // All retries exhausted - return unused budget reservation
    tr.reconcileCost(estimatedCost, 0)
    tr.updateMember(waveIdx, memberIdx, func(m *TeamMember) {
        m.Status = "failed"
    })
}
```

```
func buildPromptEnvelope(teamDir string, member *TeamMember) (string, error) {
    // Read stdin file
    stdinPath := filepath.Join(teamDir, member.StdinFile)
    stdinData, err := os.ReadFile(stdinPath)
    if err != nil {
        return "", fmt.Errorf("read stdin file: %w", err)
    }

    var stdin StdinEnvelope
    if err := json.Unmarshal(stdinData, &stdin); err != nil {
        return "", fmt.Errorf("parse stdin JSON: %w", err)
    }

    // Build prompt from template
    // (Full implementation depends on stdin schema from TC-009)

TASK: %s

CONTEXT:
%s

EXPECTED OUTPUT: Write analysis to %s following the stdout schema.

CONSTRAINTS:
- Absolute paths only
- Structured JSON output required
`,
        member.Agent,
        stdin.Task,
        stdin.Context,
        member.StdoutFile,
    )

    return envelope, nil
}
```

```
type CostError struct {
    Level     string // "warn" | "error"
    Message   string
    RawOutput string
}

func extractCostFromCLIOutput(output []byte) (float64, *CostError) {
    var result map[string]interface{}
    if err := json.Unmarshal(output, &result); err != nil {
        return 0, &CostError{
            Level:     "error",
            Message:   fmt.Sprintf("CLI output is not valid JSON: %v", err),
            RawOutput: string(output[:min(len(output), 500)]),
        }
    }
    for _, field := range []string{"cost_usd", "total_cost_usd", "usage.cost_usd"} {
        if val, ok := getNestedFloat(result, field); ok {
            return val, nil
        }
    }
    return 0, &CostError{
        Level:     "warn",
        Message:   "no cost field found in CLI JSON output",
        RawOutput: string(output[:min(len(output), 500)]),
    }
}

func getNestedFloat(m map[string]interface{}, path string) (float64, bool) {
    parts := strings.Split(path, ".")
    var current interface{} = m
    for _, part := range parts {
        if obj, ok := current.(map[string]interface{}); ok {
            current = obj[part]
        } else {
            return 0, false
        }
    }
    if f, ok := current.(float64); ok {
        return f, true
    }
    return 0, false
}
```

```
func (tr *TeamRunner) startHeartbeat() {
    heartbeatPath := filepath.Join(tr.teamDir, "heartbeat")

    go func() {
        ticker := time.NewTicker(30 * time.Second)
        defer ticker.Stop()

        for {
            select {
            case <-ticker.C:
                // Touch heartbeat file
                now := time.Now().Unix()
                os.WriteFile(heartbeatPath, []byte(fmt.Sprintf("%d\n", now)), 0644)
            case <-tr.ctx.Done():
                return
            }
        }
    }()
}
```

#### 8. Main Entry Point (`main.go`)

```
func main() {
    if len(os.Args) < 2 {
        fmt.Fprintln(os.Stderr, "Usage: gogent-team-run <team-directory>")
        os.Exit(1)
    }

    teamDir := os.Args[1]

    // Load config
    config, err := loadConfig(teamDir)
    if err != nil {
        log.Fatalf("Load config: %v", err)
    }

    // Daemon lifecycle (TC-004)
    syscall.Setsid()
    pidFile, err := acquirePIDFile(teamDir)
    if err != nil {
        log.Fatalf("PID file: %v", err)
    }
    defer pidFile.Release()

    redirectOutput(teamDir)
    daemonizeStdin()

    // Validate agents exist in agents-index.json
    if err := validateAgents(config); err != nil {
        log.Fatalf("Validate agents: %v", err)
    }

    // Update config with runner PID
    config.BackgroundPID = os.Getpid()

    // Create team runner
    ctx, cancel := context.WithCancel(context.Background())
    defer cancel()

    tr := &TeamRunner{
        teamDir:   teamDir,
        config:    config,
        childPIDs: make(map[int]struct{}),
        ctx:       ctx,
        cancel:    cancel,
    }

    // Write initial config with PID
    if err := tr.writeConfigAtomic(); err != nil {
        log.Fatalf("Write config: %v", err)
    }

    // Start heartbeat
    tr.startHeartbeat()

    // Signal handler (TC-004)
    sigCh := make(chan os.Signal, 1)
    signal.Notify(sigCh, syscall.SIGTERM, syscall.SIGINT)

    // Execute waves in goroutine
    doneCh := make(chan error, 1)
    go func() {
        doneCh <- tr.runWaves()
    }()

    // Wait for completion or signal
    select {
    case err := <-doneCh:
        if err != nil {
            log.Printf("Wave execution failed: %v", err)
            os.Exit(1)
        }
        log.Printf("All waves completed successfully")
    case sig := <-sigCh:
        log.Printf("Received signal %s, shutting down", sig)
        cancel()
        tr.killAllChildren()
        select {
        case <-doneCh:
        case <-time.After(10 * time.Second):
            log.Printf("Shutdown timeout")
        }
    }

    tr.cleanupOrphanProcesses()
}

// cleanupOrphanProcesses kills any child processes still registered after context cancellation.
func (tr *TeamRunner) cleanupOrphanProcesses() {
    tr.childrenMu.Lock()
    defer tr.childrenMu.Unlock()
    for pid := range tr.childPIDs {
        if processExists(pid) {
            log.Printf("[CLEANUP] Killing orphaned process %d", pid)
            syscall.Kill(pid, syscall.SIGKILL)
        }
    }
}
```
