### Layout dimension formulas after this change:

| Tier | Drawer Width | Drawer Height (content) | Drawer Height (minimized) |
|------|-------------|------------------------|--------------------------|
| Compact (<80) | 0 | 0 | 0 |
| Standard (80-119) | terminal width | min(cH*30/100, 12) >= 5 | 2 |
| Wide (120-179) | terminal width | min(cH*30/100, 12) >= 5 | 2 |
| Ultra (>=180) | terminal width | min(cH*30/100, 12) >= 5 | 2 |

Where cH = contentHeight. Panel heights = cH - drawerHeight.

### Code sketch — computeDrawerLayout:
```go
func (m AppModel) computeDrawerLayout(dims layoutDims) (height, width int) {
    if dims.tier == LayoutCompact || m.shared == nil || m.shared.drawerStack == nil {
        return 0, 0
    }
    drawerWidth := m.width
    hasContent := m.shared.drawerStack.OptionsHasContent() || m.shared.drawerStack.PlanHasContent()
    if hasContent {
        h := dims.contentHeight * 30 / 100
        if h > 12 { h = 12 }
        if h < 5 { h = 5 }
        if h > dims.contentHeight-5 { h = dims.contentHeight - 5 }
        if h < 1 { h = 1 }
        return h, drawerWidth
    }
    tabH := 2
    if tabH > dims.contentHeight-1 { tabH = dims.contentHeight - 1 }
    if tabH < 1 { tabH = 1 }
    return tabH, drawerWidth
}
```
