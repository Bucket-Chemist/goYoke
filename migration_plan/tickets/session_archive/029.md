# GOgent-029: Sharp Edge Schema Unification & Query API

## Description

Unify SharpEdge with the learning artifact pattern established by the 029c-f series. Extend the existing SharpEdge struct with fields for agentic query support, create QuerySharpEdges() parallel to QueryDecisions(), and provide dual output (JSONL for agents, markdown for humans).

## Design Intent

This ticket addresses the architectural gap where SharpEdge was orphaned from the query system. The 029c-f series establishes Decision/Preference/Performance as first-class queryable artifacts, but the original SharpEdge (the oldest artifact type) lacked:
1. Error message capture for full context
2. Severity classification for prioritization
3. Resolution tracking for learning loops
4. Query API for agentic retrieval

## Intended End State

- SharpEdge struct extended with ErrorMessage, Severity, Resolution fields
- SharpEdgeFilters struct for query filtering
- QuerySharpEdges(filters) function in pkg/session/query.go
- FormatSharpEdge() and FormatSharpEdges() for dual-output support
- loadPendingLearnings() continues to return []SharpEdge (already correct)
- Backward compatible with existing JSONL files
- Test coverage ≥90%

## Schema Version Strategy

**This ticket stays on schema v1.0.** Per ADR-028 and Einstein analysis:

- All new fields (`ErrorMessage`, `Severity`, `Resolution`, `ResolvedAt`) use `omitempty`
- Optional fields with `omitempty` are backward-compatible within v1.0
- Old readers (v1.0) simply ignore new fields
- New readers handle missing fields as zero values (empty string, 0)
- No migration code needed in `migrateHandoff()`
- Schema only bumps to v1.1 when 029c adds new array types to `HandoffArtifacts`

**Rationale**: JSON's additive nature means optional fields don't break existing consumers. This follows the same pattern as `Context` field which was always `omitempty`.

```go
package session

import (
    "bufio"
    "encoding/json"
    "fmt"
    "os"
    "path/filepath"
    "strings"
)

// Query provides programmatic access to session learning artifacts
type Query struct {
    ProjectDir string
}

// NewQuery creates a query instance for the given project directory
func NewQuery(projectDir string) *Query {
    return &Query{ProjectDir: projectDir}
}

// SharpEdgeFilters defines filter criteria for sharp edges
type SharpEdgeFilters struct {
    File       *string  // Glob pattern for file matching
    ErrorType  *string  // Filter by error type
    Severity   *string  // Filter by severity level (high/medium/low)
    Unresolved bool     // Only return unresolved edges (ResolvedAt == 0)
    Since      *int64   // Filter by timestamp (edges after this time)
    Limit      int      // Maximum results to return (0 = unlimited)
}

// QuerySharpEdges retrieves sharp edges with optional filters
// Returns all edges if no filters specified
// Missing file returns empty slice (not error)
//
// Example:
//   q := NewQuery("/project/dir")
//   severity := "high"
//   edges, err := q.QuerySharpEdges(SharpEdgeFilters{
//       Severity:   &severity,
//       Unresolved: true,
//   })
func (q *Query) QuerySharpEdges(filters SharpEdgeFilters) ([]SharpEdge, error) {
    edgesPath := filepath.Join(q.ProjectDir, ".claude", "memory", "pending-learnings.jsonl")

    file, err := os.Open(edgesPath)
    if err != nil {
        if os.IsNotExist(err) {
            return []SharpEdge{}, nil // Missing file is normal
        }
        return nil, fmt.Errorf("failed to open pending learnings: %w", err)
    }
    defer file.Close()

    var edges []SharpEdge
    scanner := bufio.NewScanner(file)

    for scanner.Scan() {
        line := strings.TrimSpace(scanner.Text())
        if line == "" {
            continue
        }

        var edge SharpEdge
        if err := json.Unmarshal([]byte(line), &edge); err != nil {
            continue // Skip malformed lines
        }

        // Apply filters
        if filters.File != nil && !matchGlob(edge.File, *filters.File) {
            continue
        }
        if filters.ErrorType != nil && edge.ErrorType != *filters.ErrorType {
            continue
        }
        if filters.Severity != nil && edge.Severity != *filters.Severity {
            continue
        }
        if filters.Unresolved && edge.ResolvedAt != 0 {
            continue
        }
        if filters.Since != nil && edge.Timestamp < *filters.Since {
            continue
        }

        edges = append(edges, edge)

        if filters.Limit > 0 && len(edges) >= filters.Limit {
            break
        }
    }

    if err := scanner.Err(); err != nil {
        return nil, fmt.Errorf("error reading pending learnings: %w", err)
    }

    return edges, nil
}

// matchGlob performs simple glob matching (supports * wildcard)
func matchGlob(s, pattern string) bool {
    // Simple implementation - upgrade to filepath.Match if needed
    if pattern == "*" || pattern == "" {
        return true
    }
    if strings.HasPrefix(pattern, "*") && strings.HasSuffix(pattern, "*") {
        return strings.Contains(s, pattern[1:len(pattern)-1])
    }
    if strings.HasPrefix(pattern, "*") {
        return strings.HasSuffix(s, pattern[1:])
    }
    if strings.HasSuffix(pattern, "*") {
        return strings.HasPrefix(s, pattern[:len(pattern)-1])
    }
    return s == pattern
}
```

```go
// FormatSharpEdge formats a single sharp edge as a markdown line
// Includes ErrorMessage (truncated) and Severity badge when present
func FormatSharpEdge(edge SharpEdge) string {
    // Base format: - **file**: error_type (N failures)
    formatted := fmt.Sprintf("- **%s**: %s (%d failures)",
        edge.File,
        edge.ErrorType,
        edge.ConsecutiveFailures,
    )

    // Add severity badge if present
    if edge.Severity != "" {
        badge := map[string]string{
            "high":   "🔴",
            "medium": "🟡",
            "low":    "🟢",
        }[edge.Severity]
        if badge == "" {
            badge = "⚪"
        }
        formatted = badge + " " + formatted
    }

    // Add error message if present (truncated to 100 chars)
    if edge.ErrorMessage != "" {
        truncated := edge.ErrorMessage
        if len(truncated) > 100 {
            truncated = truncated[:100] + "..."
        }
        formatted += fmt.Sprintf("\n  Error: `%s`", truncated)
    }

    // Add resolution if present
    if edge.Resolution != "" {
        formatted += fmt.Sprintf("\n  ✅ Resolved: %s", edge.Resolution)
    }

    return formatted
}

// FormatSharpEdges formats multiple sharp edges as a markdown section
func FormatSharpEdges(edges []SharpEdge) string {
    if len(edges) == 0 {
        return ""
    }

    var sb strings.Builder
    sb.WriteString("## Sharp Edges\n\n")

    for _, edge := range edges {
        sb.WriteString(FormatSharpEdge(edge))
        sb.WriteString("\n\n")
    }

    return sb.String()
}
```

```go
// ValidateSharpEdge validates SharpEdge JSON against schema
// FIXED: Field names now match struct (consecutive_failures, timestamp)
func ValidateSharpEdge(data []byte) error {
    var edge map[string]interface{}
    if err := json.Unmarshal(data, &edge); err != nil {
        return fmt.Errorf("[sharp-edge-validation] Invalid JSON: %w", err)
    }

    // Required: file
    if file, ok := edge["file"].(string); !ok || file == "" {
        return fmt.Errorf("[sharp-edge-validation] Missing required field 'file'")
    }

    // Required: error_type
    if errType, ok := edge["error_type"].(string); !ok || errType == "" {
        return fmt.Errorf("[sharp-edge-validation] Missing required field 'error_type'")
    }

    // Required: consecutive_failures (minimum 3)
    // FIXED: Was incorrectly checking "failure_count"
    var count int
    switch v := edge["consecutive_failures"].(type) {
    case float64:
        count = int(v)
    case int:
        count = v
    default:
        return fmt.Errorf("[sharp-edge-validation] Missing or invalid 'consecutive_failures'")
    }
    if count < 3 {
        return fmt.Errorf("[sharp-edge-validation] consecutive_failures %d < 3", count)
    }

    // Required: timestamp
    // FIXED: Was incorrectly checking "last_occurrence"
    switch v := edge["timestamp"].(type) {
    case float64:
        if int64(v) == 0 {
            return fmt.Errorf("[sharp-edge-validation] Missing 'timestamp'")
        }
    case int64, int:
        // Valid
    default:
        return fmt.Errorf("[sharp-edge-validation] Invalid 'timestamp' type")
    }

    // Optional fields validated when present
    if severity, exists := edge["severity"]; exists {
        if s, ok := severity.(string); ok {
            valid := map[string]bool{"high": true, "medium": true, "low": true}
            if !valid[s] {
                return fmt.Errorf("[sharp-edge-validation] Invalid severity '%s'", s)
            }
        }
    }

    return nil
}
```
