const HandoffSchemaVersion = "1.0"

func LoadHandoff(path string) (*Handoff, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, fmt.Errorf("[handoff] Failed to read JSONL from %s: %w. Check file exists and is readable.", path, err)
	}

	lines := bytes.Split(data, []byte("\n"))
	if len(lines) == 0 {
		return nil, fmt.Errorf("[handoff] Empty JSONL file at %s. Expected at least one handoff entry.", path)
	}

	// Parse last non-empty line
	var lastLine []byte
	for i := len(lines) - 1; i >= 0; i-- {
		if len(bytes.TrimSpace(lines[i])) > 0 {
			lastLine = lines[i]
			break
		}
	}

	if lastLine == nil {
		return nil, fmt.Errorf("[handoff] No valid JSONL entries in %s. File may be corrupted.", path)
	}

	// Check schema version first
	var versionCheck struct {
		SchemaVersion string `json:"schema_version"`
	}
	if err := json.Unmarshal(lastLine, &versionCheck); err != nil {
		return nil, fmt.Errorf("[handoff] Failed to parse JSONL schema version from %s: %w. File may be corrupted.", path, err)
	}

	// Migrate if needed
	if versionCheck.SchemaVersion != HandoffSchemaVersion {
		return migrateHandoff(versionCheck.SchemaVersion, lastLine)
	}

	// Current version - parse directly
	var handoff Handoff
	if err := json.Unmarshal(lastLine, &handoff); err != nil {
		return nil, fmt.Errorf("[handoff] Failed to parse JSONL from %s: %w. Schema may have changed.", path, err)
	}

	return &handoff, nil
}

// migrateHandoff handles conversion from older schema versions to current
func migrateHandoff(oldVersion string, data []byte) (*Handoff, error) {
	switch oldVersion {
	case "1.0":
		// Already current version - no migration needed
		var handoff Handoff
		if err := json.Unmarshal(data, &handoff); err != nil {
			return nil, fmt.Errorf("[handoff] Failed to parse v1.0 handoff: %w", err)
		}
		return &handoff, nil

	case "2.0":
		// Future version - migration stub
		// When v2.0 is defined, implement conversion logic here
		return nil, fmt.Errorf("[handoff] Migration from v2.0 to v%s not yet implemented. This is a bug.", HandoffSchemaVersion)

	default:
		return nil, fmt.Errorf("[handoff] Unsupported schema version %s. Expected v%s or older. Upgrade gogent-archive binary.", oldVersion, HandoffSchemaVersion)
	}
}

func TestLoadHandoff_CurrentVersion(t *testing.T) {
	tmpDir := t.TempDir()
	handoffPath := filepath.Join(tmpDir, "handoffs.jsonl")

	// Write v1.0 handoff
	handoff := &Handoff{
		SessionID:      "test-v1",
		Timestamp:      time.Now().Unix(),
		SchemaVersion:  "1.0",
		SessionMetrics: &SessionMetrics{ToolCalls: 5},
	}

	jsonData, _ := json.Marshal(handoff)
	os.WriteFile(handoffPath, jsonData, 0644)

	loaded, err := LoadHandoff(handoffPath)

	if err != nil {
		t.Fatalf("Expected no error loading v1.0, got: %v", err)
	}

	if loaded.SchemaVersion != "1.0" {
		t.Errorf("Expected schema version 1.0, got: %s", loaded.SchemaVersion)
	}

	if loaded.SessionID != "test-v1" {
		t.Errorf("Expected session ID test-v1, got: %s", loaded.SessionID)
	}
}
