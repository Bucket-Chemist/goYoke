### Implementation Pattern
```go
func ClusterViolationsByType(violations []routing.Violation) map[string]*ViolationCluster {
	clusters := make(map[string]*ViolationCluster)

	for _, v := range violations {
		cluster, exists := clusters[v.ViolationType]
		if !exists {
			cluster = &ViolationCluster{
				Type:    v.ViolationType,
				Count:   0,
				Samples: []routing.Violation{},
			}
			clusters[v.ViolationType] = cluster
		}

		cluster.Count++

		// Keep first 3 as samples
		if len(cluster.Samples) < 3 {
			cluster.Samples = append(cluster.Samples, v)
		}
	}

	return clusters
}
```
