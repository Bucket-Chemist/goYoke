// DefaultHandoffConfig creates default paths for handoff generation
func DefaultHandoffConfig(projectDir string) *HandoffConfig {
	claudeDir := filepath.Join(projectDir, ".claude", "memory")

	return &HandoffConfig{
		ProjectDir:     projectDir,
		HandoffPath:    filepath.Join(claudeDir, "last-handoff.md"),
		PendingPath:    filepath.Join(claudeDir, "pending-learnings.jsonl"),
		ViolationsPath: filepath.Join(claudeDir, "routing-violations.jsonl"),
	}
}

// GenerateHandoff creates a markdown handoff document with session context
func GenerateHandoff(config *HandoffConfig, metrics *SessionMetrics) error {
	if config == nil {
		return fmt.Errorf("[handoff] Config nil. Cannot generate handoff without configuration. Provide valid HandoffConfig.")
	}

	if metrics == nil {
		return fmt.Errorf("[handoff] Metrics nil. Cannot generate handoff without session metrics. Provide valid SessionMetrics.")
	}

	// Ensure directory exists
	dir := filepath.Dir(config.HandoffPath)
	if err := os.MkdirAll(dir, 0755); err != nil {
		return fmt.Errorf("[handoff] Failed to create directory %s: %w. Check write permissions.", dir, err)
	}

	// Create handoff file
	f, err := os.Create(config.HandoffPath)
	if err != nil {
		return fmt.Errorf("[handoff] Failed to create file %s: %w. Check write permissions.", config.HandoffPath, err)
	}
	defer f.Close()

	// Write header
	timestamp := time.Now().Format("2006-01-02 15:04:05")
	fmt.Fprintf(f, "# Session Handoff - %s\n\n", timestamp)

	// Write session metrics
	fmt.Fprintf(f, "## Session Metrics\n")
	fmt.Fprintf(f, "- **Tool Calls**: ~%d\n", metrics.ToolCallCount)
	fmt.Fprintf(f, "- **Errors Logged**: %d\n", metrics.ErrorCount)
	fmt.Fprintf(f, "- **Routing Violations**: %d\n", metrics.ViolationCount)
	fmt.Fprintf(f, "- **Session ID**: %s\n\n", metrics.SessionID)

	// Write pending learnings
	if err := writePendingLearnings(f, config.PendingPath); err != nil {
		return err
	}

	// Write violations summary
	if err := writeViolationsSummary(f, config.ViolationsPath); err != nil {
		return err
	}

	// Write context and action items
	fmt.Fprintf(f, "## Context for Next Session\n")
	fmt.Fprintf(f, "1. Review pending learnings before continuing work\n")
	fmt.Fprintf(f, "2. Check if any sharp edges should be added to agent sharp-edges.yaml\n")
	fmt.Fprintf(f, "3. Validate routing decisions match patterns from last session\n\n")

	fmt.Fprintf(f, "## Immediate Actions\n")
	fmt.Fprintf(f, "- [ ] Process pending sharp edges\n")
	fmt.Fprintf(f, "- [ ] Review routing violations for pattern issues\n")
	fmt.Fprintf(f, "- [ ] Update TODO if tasks remain incomplete\n")

	return nil
}

// writePendingLearnings reads pending learnings JSONL and formats for handoff
func writePendingLearnings(f *os.File, pendingPath string) error {
	fmt.Fprintf(f, "## Pending Learnings\n")

	// Check if file exists
	if _, err := os.Stat(pendingPath); os.IsNotExist(err) {
		fmt.Fprintf(f, "No sharp edges captured this session.\n\n")
		return nil
	}

	// Read file
	data, err := os.ReadFile(pendingPath)
	if err != nil {
		return fmt.Errorf("[handoff] Failed to read pending learnings from %s: %w. Check file permissions.", pendingPath, err)
	}

	if len(data) == 0 {
		fmt.Fprintf(f, "No sharp edges captured this session.\n\n")
		return nil
	}

	// Write raw JSONL for now (GOgent-029 will add formatting)
	fmt.Fprintf(f, "```jsonl\n%s\n```\n\n", string(data))

	return nil
}

// writeViolationsSummary reads violations JSONL and formats for handoff
func writeViolationsSummary(f *os.File, violationsPath string) error {
	// Check if file exists
	if _, err := os.Stat(violationsPath); os.IsNotExist(err) {
		// No violations file = no violations, skip section entirely
		return nil
	}

	// Read file
	data, err := os.ReadFile(violationsPath)
	if err != nil {
		return fmt.Errorf("[handoff] Failed to read violations from %s: %w. Check file permissions.", violationsPath, err)
	}

	if len(data) == 0 {
		// Empty file = no violations, skip section
		return nil
	}

	// Write violations section
	fmt.Fprintf(f, "## Routing Violations\n")
	fmt.Fprintf(f, "```jsonl\n%s\n```\n\n", string(data))

	return nil
}

func TestGenerateHandoff_CompleteHandoff(t *testing.T) {
	tmpDir := t.TempDir()
	config := DefaultHandoffConfig(tmpDir)

	// Create pending learnings file
	pendingData := `{"file":"test.go","error_type":"nil_pointer","consecutive_failures":3}`
	os.MkdirAll(filepath.Dir(config.PendingPath), 0755)

	// Create violations file

	os.WriteFile(config.ViolationsPath, []byte(violationsData), 0644)

	metrics := &SessionMetrics{
		ToolCallCount:  42,
		ErrorCount:     5,
		ViolationCount: 2,
		SessionID:      "session-123",
	}

	err := GenerateHandoff(config, metrics)

	if err != nil {
		t.Fatalf("Expected no error, got: %v", err)
	}

	// Verify file was created
	if _, err := os.Stat(config.HandoffPath); os.IsNotExist(err) {
		t.Fatal("Handoff file was not created")
	}

	// Read and verify content
	content, err := os.ReadFile(config.HandoffPath)
	if err != nil {
		t.Fatalf("Failed to read handoff file: %v", err)
	}

	contentStr := string(content)

	// Verify required sections
	requiredSections := []string{
		"# Session Handoff",
		"## Session Metrics",
		"Tool Calls**: ~42",
		"Errors Logged**: 5",
		"Routing Violations**: 2",
		"Session ID**: session-123",
		"## Pending Learnings",
		"## Routing Violations",
		"## Context for Next Session",
		"## Immediate Actions",
	}

	for _, section := range requiredSections {
		if !strings.Contains(contentStr, section) {
			t.Errorf("Missing required section: %s", section)
		}
	}

	// Verify pending learnings data included
	if !strings.Contains(contentStr, "nil_pointer") {
		t.Error("Pending learnings data not included")
	}

	// Verify violations data included
	if !strings.Contains(contentStr, "test-agent") {
		t.Error("Violations data not included")
	}
}
