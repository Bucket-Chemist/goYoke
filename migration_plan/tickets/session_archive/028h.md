## Implementation Details

### Implementation
Replace stub in `pkg/session/handoff.go`:

```go
import "os/exec"

func collectGitInfo(projectDir string) GitInfo {
	info := GitInfo{}

	// Check if git repo
	cmd := exec.Command("git", "rev-parse", "--is-inside-work-tree")
	cmd.Dir = projectDir
	if err := cmd.Run(); err != nil {
		// Not a git repo - return empty info
		return info
	}

	// Get current branch
	cmd = exec.Command("git", "rev-parse", "--abbrev-ref", "HEAD")
	cmd.Dir = projectDir
	output, err := cmd.Output()
	if err == nil {
		info.Branch = strings.TrimSpace(string(output))
	}

	// Get dirty status
	cmd = exec.Command("git", "status", "--porcelain")
	cmd.Dir = projectDir
	output, err = cmd.Output()
	if err == nil {
		info.HasUncommittedChanges = len(strings.TrimSpace(string(output))) > 0
	}

	return info
}
```

```go
func TestCollectGitInfo_ValidRepo(t *testing.T) {
	// This test requires actual git repo
	// Skip if not in git environment
	cmd := exec.Command("git", "rev-parse", "--is-inside-work-tree")
	if err := cmd.Run(); err != nil {
		t.Skip("Not in git repo, skipping git info test")
	}

	info := collectGitInfo(".")
	if info.Branch == "" {
		t.Error("Expected branch name in git repo")
	}
}

func TestCollectGitInfo_NonGitDir(t *testing.T) {
	tmpDir := t.TempDir()

	info := collectGitInfo(tmpDir)
	if info.Branch != "" || info.HasUncommittedChanges {
		t.Error("Expected empty GitInfo for non-git directory")
	}
}
```
