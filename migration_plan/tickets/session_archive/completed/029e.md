```go
// DecisionFilters defines filter criteria for decisions
type DecisionFilters struct {
	Category *string  // Optional category filter (e.g., "architecture")
	Impact   *string  // Optional impact filter (e.g., "high")
	Since    *int64   // Optional Unix timestamp filter
	Limit    int      // Maximum results to return (0 = unlimited)
}

// PreferenceFilters defines filter criteria for preferences
type PreferenceFilters struct {
	Category *string  // Optional category filter (e.g., "routing")
	Scope    *string  // Optional scope filter (e.g., "project")
	Since    *int64   // Optional Unix timestamp filter
	Limit    int      // Maximum results to return (0 = unlimited)
}

// PerformanceFilters defines filter criteria for performance metrics
type PerformanceFilters struct {
	Operation   *string  // Optional operation filter (e.g., "handoff_generation")
	MinDuration *int64   // Optional minimum duration filter in milliseconds
	OnlyFailed  bool     // If true, only return failed operations
	Since       *int64   // Optional Unix timestamp filter
	Limit       int      // Maximum results to return (0 = unlimited)
}
```

### Query Implementations

#### QueryDecisions

```go
// QueryDecisions retrieves decisions from memory with optional filters
// Returns all decisions if no filters specified
// Example:
//   q := NewQuery("/project/dir")
//   category := "architecture"
//   impact := "high"
//   decisions, err := q.QueryDecisions(DecisionFilters{
//       Category: &category,
//       Impact: &impact,
//   })
func (q *Query) QueryDecisions(filters DecisionFilters) ([]Decision, error) {
	decisionsPath := filepath.Join(q.ProjectDir, ".claude", "memory", "decisions.jsonl")

	file, err := os.Open(decisionsPath)
	if err != nil {
		if os.IsNotExist(err) {
			return []Decision{}, nil // Missing file is normal
		}
		return nil, fmt.Errorf("failed to open decisions file: %w", err)
	}
	defer file.Close()

	var decisions []Decision
	scanner := bufio.NewScanner(file)

	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line == "" {
			continue
		}

		var d Decision
		if err := json.Unmarshal([]byte(line), &d); err != nil {
			continue // Skip malformed lines
		}

		// Apply filters
		if filters.Category != nil && d.Category != *filters.Category {
			continue
		}
		if filters.Impact != nil && d.Impact != *filters.Impact {
			continue
		}
		if filters.Since != nil && d.Timestamp < *filters.Since {
			continue
		}

		decisions = append(decisions, d)

		// Apply limit
		if filters.Limit > 0 && len(decisions) >= filters.Limit {
			break
		}
	}

	if err := scanner.Err(); err != nil {
		return nil, fmt.Errorf("error reading decisions: %w", err)
	}

	return decisions, nil
}
```

#### QueryPerformance

```go
// QueryPerformance retrieves performance metrics from memory with optional filters
// Returns all metrics if no filters specified
// Example:
//   q := NewQuery("/project/dir")
//   minDuration := int64(1000) // 1 second
//   metrics, err := q.QueryPerformance(PerformanceFilters{
//       MinDuration: &minDuration,
//       OnlyFailed:  false,
//   })
func (q *Query) QueryPerformance(filters PerformanceFilters) ([]PerformanceMetric, error) {
	performancePath := filepath.Join(q.ProjectDir, ".claude", "memory", "performance.jsonl")

	file, err := os.Open(performancePath)
	if err != nil {
		if os.IsNotExist(err) {
			return []PerformanceMetric{}, nil
		}
		return nil, fmt.Errorf("failed to open performance file: %w", err)
	}
	defer file.Close()

	var metrics []PerformanceMetric
	scanner := bufio.NewScanner(file)

	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line == "" {
			continue
		}

		var m PerformanceMetric
		if err := json.Unmarshal([]byte(line), &m); err != nil {
			continue
		}

		// Apply filters
		if filters.Operation != nil && m.Operation != *filters.Operation {
			continue
		}
		if filters.MinDuration != nil && m.DurationMs < *filters.MinDuration {
			continue
		}
		if filters.OnlyFailed && m.Success {
			continue // Skip successful operations when only_failed is true
		}
		if filters.Since != nil && m.Timestamp < *filters.Since {
			continue
		}

		metrics = append(metrics, m)

		if filters.Limit > 0 && len(metrics) >= filters.Limit {
			break
		}
	}

	if err := scanner.Err(); err != nil {
		return nil, fmt.Errorf("error reading performance: %w", err)
	}

	return metrics, nil
}
```

## Why This Matters
Programmatic query API enables:
- **Agent integration**: systems-architecture-reviewer can retrieve decisions without reading full JSONL
- **Context efficiency**: Filters reduce token usage (return only relevant learnings)
- **Separation of concerns**: CLI uses same API (single source of truth)
- **Composability**: Agents can combine queries (e.g., high-impact architecture decisions from last week)
- **Future extensibility**: Foundation for advanced features (trending, correlation analysis)

Critical for transforming learnings from "archived data" to "queryable knowledge" for agent workflows.
