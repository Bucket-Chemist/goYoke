# GOgent-028j: JSONL History Querying

## Description
Extend gogent-archive with subcommands: `list`, `show <session-id>`, `stats`. Enables retrospective analysis of session history without manual JSONL parsing.

## Implementation Intention
Make handoff history queryable for humans (debugging, retrospectives).

## Intended End State
- `gogent-archive list` - shows all sessions
- `gogent-archive show <session-id>` - renders specific handoff as markdown
- `gogent-archive stats` - aggregate metrics across sessions
- Useful for team retrospectives

## Dependencies
- GOgent-028: Handoff generation must exist

### Subcommand Structure
Update `cmd/gogent-archive/main.go`:

```go
package main

import (
	"flag"
	"fmt"
	"os"
)

func main() {
	if len(os.Args) > 1 {
		switch os.Args[1] {
		case "list":
			listSessions()
			return
		case "show":
			if len(os.Args) < 3 {
				fmt.Fprintln(os.Stderr, "[gogent-archive] Usage: gogent-archive show <session-id>")
				os.Exit(1)
			}
			showSession(os.Args[2])
			return
		case "stats":
			showStats()
			return
		case "--help", "-h":
			printHelp()
			return
		}
	}

	// Default: SessionEnd hook mode
	if err := run(); err != nil {
		outputError(err.Error())
		os.Exit(1)
	}
}

func listSessions() {
	projectDir := getProjectDir()
	handoffPath := filepath.Join(projectDir, ".claude", "memory", "handoffs.jsonl")

	handoffs, err := session.LoadAllHandoffs(handoffPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to load handoffs: %v\n", err)
		os.Exit(1)
	}

	// Print table header
	fmt.Println("Session ID                    | Timestamp  | Tool Calls | Errors | Violations")
	fmt.Println("------------------------------|------------|------------|--------|------------")

	for _, h := range handoffs {
		timestamp := time.Unix(h.Timestamp, 0).Format("2006-01-02")
		fmt.Printf("%-30s | %-10s | %10d | %6d | %10d\n",
			h.SessionID, timestamp, h.Context.Metrics.ToolCalls,
			h.Context.Metrics.ErrorsLogged, h.Context.Metrics.RoutingViolations)
	}
}

func showSession(sessionID string) {
	projectDir := getProjectDir()
	handoffPath := filepath.Join(projectDir, ".claude", "memory", "handoffs.jsonl")

	handoffs, err := session.LoadAllHandoffs(handoffPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to load handoffs: %v\n", err)
		os.Exit(1)
	}

	for _, h := range handoffs {
		if h.SessionID == sessionID {
			// Render this handoff as markdown to stdout
			if err := session.RenderMarkdownToWriter(h, os.Stdout); err != nil {
				fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to render handoff: %v\n", err)
				os.Exit(1)
			}
			return
		}
	}

	fmt.Fprintf(os.Stderr, "[gogent-archive] Session %s not found in handoff history.\n", sessionID)
	os.Exit(1)
}

func showStats() {
	projectDir := getProjectDir()
	handoffPath := filepath.Join(projectDir, ".claude", "memory", "handoffs.jsonl")

	handoffs, err := session.LoadAllHandoffs(handoffPath)
	if err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to load handoffs: %v\n", err)
		os.Exit(1)
	}

	totalSessions := len(handoffs)
	totalToolCalls := 0
	totalErrors := 0
	totalViolations := 0

	for _, h := range handoffs {
		totalToolCalls += h.Context.Metrics.ToolCalls
		totalErrors += h.Context.Metrics.ErrorsLogged
		totalViolations += h.Context.Metrics.RoutingViolations
	}

	avgToolCalls := 0
	if totalSessions > 0 {
		avgToolCalls = totalToolCalls / totalSessions
	}

	fmt.Printf("Total Sessions: %d\n", totalSessions)
	fmt.Printf("Avg Tool Calls per Session: %d\n", avgToolCalls)
	fmt.Printf("Total Errors: %d\n", totalErrors)
	fmt.Printf("Total Violations: %d\n", totalViolations)
}

func printHelp() {
	fmt.Println("gogent-archive - Session handoff archival and querying")
	fmt.Println("")
	fmt.Println("Usage:")
	fmt.Println("  gogent-archive              Read SessionEnd JSON from STDIN (hook mode)")
	fmt.Println("  gogent-archive list         List all sessions")
	fmt.Println("  gogent-archive show <id>    Show specific session handoff")
	fmt.Println("  gogent-archive stats        Show aggregate statistics")
	fmt.Println("  gogent-archive --help       Show this help")
}
```

### New Helper Function
Add to `pkg/session/handoff_markdown.go`:

```go
// RenderMarkdownToWriter renders handoff to io.Writer instead of file
func RenderMarkdownToWriter(handoff *Handoff, w io.Writer) error {
	content := generateMarkdownContent(handoff)
	_, err := w.Write([]byte(content))
	return err
}
```
