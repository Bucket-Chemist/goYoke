### Implementation
```go
package routing

import (
	"strings"
)

// WIPContext represents work-in-progress state from last edit
type WIPContext struct {
	FilePath    string
	LineContext string // First line of old_string from Edit, or first line from Write
	Timestamp   int64
	ToolUsed    string // "Edit" or "Write"
}

// DetectWorkInProgress finds the last edit/write operation for resume guidance
func DetectWorkInProgress(events []ToolEvent) *WIPContext {
	// Search backwards from end of events
	for i := len(events) - 1; i >= 0; i-- {
		event := events[i]

		if event.ToolName == "Edit" || event.ToolName == "Write" {
			filePath, _ := event.ToolInput["file_path"].(string)
			if filePath == "" {
				continue
			}

			context := ""
			if event.ToolName == "Edit" {
				// Extract first line of old_string
				if oldStr, ok := event.ToolInput["old_string"].(string); ok {
					lines := strings.Split(oldStr, "\n")
					if len(lines) > 0 {
						context = lines[0]
					}
				}
			} else if event.ToolName == "Write" {
				// Extract first line of content
				if content, ok := event.ToolInput["content"].(string); ok {
					lines := strings.Split(content, "\n")
					if len(lines) > 0 {
						context = lines[0]
					}
				}
			}

			return &WIPContext{
				FilePath:    filePath,
				LineContext: context,
				Timestamp:   event.Timestamp,
				ToolUsed:    event.ToolName,
			}
		}
	}

	// No Edit/Write events found
	return nil
}
```

package routing

import (

func TestDetectWorkInProgress_EndsWithEdit(t *testing.T) {
	events := []ToolEvent{
		{ToolName: "Read", Timestamp: 1000},
		{ToolName: "Edit", Timestamp: 1100, ToolInput: map[string]interface{}{
			"file_path":  "test.go",
			"old_string": "func main() {\n\treturn\n}",
		}},
	}

	result := DetectWorkInProgress(events)

	if result == nil {
		t.Fatal("Expected WIPContext, got nil")
	}

	if result.FilePath != "test.go" {
		t.Errorf("Expected FilePath 'test.go', got: %s", result.FilePath)
	}

	if result.LineContext != "func main() {" {
		t.Errorf("Expected LineContext 'func main() {', got: %s", result.LineContext)
	}

	if result.ToolUsed != "Edit" {
		t.Errorf("Expected ToolUsed 'Edit', got: %s", result.ToolUsed)
	}
}
