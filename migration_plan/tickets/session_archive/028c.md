# GOgent-028c: Implement DetectWorkInProgress()

## Description
Detect if session ended mid-task by identifying the last Edit/Write operation and extracting context.

### Implementation Pattern
```go
func DetectWorkInProgress(events []ToolEvent) *WIPContext {
	// Search backwards from end of events
	for i := len(events) - 1; i >= 0; i-- {
		event := events[i]

		if event.ToolName == "Edit" || event.ToolName == "Write" {
			filePath, _ := event.ToolInput["file_path"].(string)
			if filePath == "" {
				continue
			}

			context := ""
			if event.ToolName == "Edit" {
				// Extract first line of old_string
				if oldStr, ok := event.ToolInput["old_string"].(string); ok {
					lines := strings.Split(oldStr, "\n")
					if len(lines) > 0 {
						context = lines[0]
					}
				}
			} else if event.ToolName == "Write" {
				// Extract first line of content
				if content, ok := event.ToolInput["content"].(string); ok {
					lines := strings.Split(content, "\n")
					if len(lines) > 0 {
						context = lines[0]
					}
				}
			}

			return &WIPContext{
				FilePath:    filePath,
				LineContext: context,
				Timestamp:   event.Timestamp,
				ToolUsed:    event.ToolName,
			}
		}
	}

	// No Edit/Write events found
	return nil
}
```
