// listDecisions displays architectural decisions with optional filtering
func listDecisions() {
	decisionsFlags := flag.NewFlagSet("decisions", flag.ExitOnError)
	categoryFlag := decisionsFlags.String("category", "", "Filter by category (architecture, tooling, pattern)")
	impactFlag := decisionsFlags.String("impact", "", "Filter by impact (high, medium, low)")
	sinceFlag := decisionsFlags.String("since", "", "Filter since date (YYYY-MM-DD) or duration (7d)")
	decisionsFlags.Parse(os.Args[2:])

	projectDir := getProjectDir()
	decisionsPath := filepath.Join(projectDir, ".claude", "memory", "decisions.jsonl")

	file, err := os.Open(decisionsPath)
	if err != nil {
		if os.IsNotExist(err) {
			fmt.Println("No decisions recorded. Run Claude Code in this project to generate decision history.")
			return
		}
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to open decisions file: %v\n", err)
		fmt.Fprintln(os.Stderr, "  Verify .claude/memory/decisions.jsonl exists and is readable.")
		os.Exit(1)
	}
	defer file.Close()

	// Parse all decisions
	var decisions []session.Decision
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line == "" {
			continue
		}

		var d session.Decision
		if err := json.Unmarshal([]byte(line), &d); err != nil {
			continue // Skip malformed lines
		}
		decisions = append(decisions, d)
	}

	if err := scanner.Err(); err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Error reading decisions: %v\n", err)
		os.Exit(1)
	}

	if len(decisions) == 0 {
		fmt.Println("No decisions recorded.")
		return
	}

	// Apply filters
	if *categoryFlag != "" {
		decisions = filterDecisionsByCategory(decisions, *categoryFlag)
	}
	if *impactFlag != "" {
		decisions = filterDecisionsByImpact(decisions, *impactFlag)
	}
	if *sinceFlag != "" {
		decisions = filterDecisionsSince(decisions, *sinceFlag)
	}

	if len(decisions) == 0 {
		fmt.Println("No decisions match the specified filters.")
		return
	}

	// Print table header
	fmt.Println("Timestamp  | Category      | Impact | Decision                           | Rationale")
	fmt.Println("-----------|---------------|--------|------------------------------------|--------------------------")

	for _, d := range decisions {
		timestamp := time.Unix(d.Timestamp, 0).Format("2006-01-02")
		decision := truncate(d.Decision, 35)
		rationale := truncate(d.Rationale, 25)
		fmt.Printf("%s | %-13s | %-6s | %-35s | %s\n",
			timestamp, d.Category, d.Impact, decision, rationale)
	}

	fmt.Printf("\nTotal: %d decision(s)\n", len(decisions))
}

// filterDecisionsByCategory filters decisions by category
func filterDecisionsByCategory(decisions []session.Decision, category string) []session.Decision {
	var filtered []session.Decision
	for _, d := range decisions {
		if d.Category == category {
			filtered = append(filtered, d)
		}
	}
	return filtered
}

// filterDecisionsByImpact filters decisions by impact level
func filterDecisionsByImpact(decisions []session.Decision, impact string) []session.Decision {
	var filtered []session.Decision
	for _, d := range decisions {
		if d.Impact == impact {
			filtered = append(filtered, d)
		}
	}
	return filtered
}

// filterDecisionsSince filters decisions by timestamp
func filterDecisionsSince(decisions []session.Decision, since string) []session.Decision {
	cutoff := parseSinceFilter(since)
	var filtered []session.Decision
	for _, d := range decisions {
		if time.Unix(d.Timestamp, 0).After(cutoff) || time.Unix(d.Timestamp, 0).Equal(cutoff) {
			filtered = append(filtered, d)
		}
	}
	return filtered
}

// parseSinceFilter parses --since flag (duration like "7d" or date like "2026-01-15")
func parseSinceFilter(since string) time.Time {
	// Try duration format first
	if strings.HasSuffix(since, "d") {
		daysStr := strings.TrimSuffix(since, "d")
		days, err := strconv.Atoi(daysStr)
		if err == nil {
			return time.Now().AddDate(0, 0, -days)
		}
	}

	// Try date format
	parsedDate, err := time.Parse("2006-01-02", since)
	if err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Invalid --since format '%s'\n", since)
		fmt.Fprintln(os.Stderr, "  Use duration (e.g., '7d') or date (YYYY-MM-DD)")
		os.Exit(1)
	}
	return parsedDate
}

// truncate truncates string to maxLen with ellipsis
func truncate(s string, maxLen int) string {
	if len(s) <= maxLen {
		return s
	}
	return s[:maxLen-3] + "..."
}

	categoryFlag := preferencesFlags.String("category", "", "Filter by category (routing, tooling, formatting)")
	scopeFlag := preferencesFlags.String("scope", "", "Filter by scope (session, project, global)")

	projectDir := getProjectDir()
	preferencesPath := filepath.Join(projectDir, ".claude", "memory", "preferences.jsonl")

	file, err := os.Open(preferencesPath)
	if err != nil {
		if os.IsNotExist(err) {
			fmt.Println("No preferences recorded.")
			return
		}
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to open preferences file: %v\n", err)
		os.Exit(1)

	defer file.Close()

	// Parse all preferences
	var preferences []session.PreferenceOverride
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())

			continue
		}

		if err := json.Unmarshal([]byte(line), &p); err != nil {
			continue
		}
		preferences = append(preferences, p)
	}

	if err := scanner.Err(); err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Error reading preferences: %v\n", err)
		os.Exit(1)
	}

	if len(preferences) == 0 {
		fmt.Println("No preferences recorded.")
		return
	}

	// Apply filters
	if *categoryFlag != "" {
		preferences = filterPreferencesByCategory(preferences, *categoryFlag)
	}
	if *scopeFlag != "" {
		preferences = filterPreferencesByScope(preferences, *scopeFlag)
	}
	if *sinceFlag != "" {
		preferences = filterPreferencesSince(preferences, *sinceFlag)
	}

	if len(preferences) == 0 {
		fmt.Println("No preferences match the specified filters.")
		return
	}

	// Print table header
	fmt.Println("Timestamp  | Category  | Scope   | Key                  | Value        | Reason")
	fmt.Println("-----------|-----------|---------|----------------------|--------------|--------------------")

	for _, p := range preferences {
		timestamp := time.Unix(p.Timestamp, 0).Format("2006-01-02")
		key := truncate(p.Key, 20)
		value := truncate(p.Value, 12)
		reason := truncate(p.Reason, 19)
		fmt.Printf("%s | %-9s | %-7s | %-20s | %-12s | %s\n",
			timestamp, p.Category, p.Scope, key, value, reason)
	}

	fmt.Printf("\nTotal: %d preference(s)\n", len(preferences))
}

// showPerformance displays performance metrics with optional grouping
func showPerformance() {
	performanceFlags := flag.NewFlagSet("performance", flag.ExitOnError)
	byOperation := performanceFlags.Bool("by-operation", false, "Group metrics by operation type")
	slowOnly := performanceFlags.Bool("slow-only", false, "Show only metrics >1000ms")
	sinceFlag := performanceFlags.String("since", "", "Filter since date (YYYY-MM-DD) or duration (7d)")
	performanceFlags.Parse(os.Args[2:])

	projectDir := getProjectDir()
	performancePath := filepath.Join(projectDir, ".claude", "memory", "performance.jsonl")

	file, err := os.Open(performancePath)
	if err != nil {
		if os.IsNotExist(err) {
			fmt.Println("No performance metrics recorded.")
			return
		}
		fmt.Fprintf(os.Stderr, "[gogent-archive] Failed to open performance file: %v\n", err)
		os.Exit(1)
	}
	defer file.Close()

	// Parse all metrics
	var metrics []session.PerformanceMetric
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line == "" {
			continue
		}

		var m session.PerformanceMetric
		if err := json.Unmarshal([]byte(line), &m); err != nil {
			continue
		}
		metrics = append(metrics, m)
	}

	if err := scanner.Err(); err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-archive] Error reading performance: %v\n", err)
		os.Exit(1)
	}

	if len(metrics) == 0 {
		fmt.Println("No performance metrics recorded.")
		return
	}

	// Apply filters
	if *slowOnly {
		metrics = filterPerformanceSlow(metrics)
	}
	if *sinceFlag != "" {
		metrics = filterPerformanceSince(metrics, *sinceFlag)
	}

	if len(metrics) == 0 {
		fmt.Println("No performance metrics match the specified filters.")
		return
	}

	// Aggregate stats
	totalMetrics := len(metrics)
	totalDuration := int64(0)
	failedCount := 0
	byOperationMap := make(map[string]int64)
	byOperationCount := make(map[string]int)

	for _, m := range metrics {
		totalDuration += m.DurationMs
		if !m.Success {
			failedCount++
		}
		if *byOperation {
			byOperationMap[m.Operation] += m.DurationMs
			byOperationCount[m.Operation]++
		}
	}

	// Print summary
	avgDuration := int64(0)
	if totalMetrics > 0 {
		avgDuration = totalDuration / int64(totalMetrics)
	}

	fmt.Printf("Total Metrics: %d\n", totalMetrics)
	fmt.Printf("Total Duration: %dms\n", totalDuration)
	fmt.Printf("Avg Duration: %dms\n", avgDuration)
	fmt.Printf("Failed Operations: %d (%.1f%%)\n", failedCount, float64(failedCount)/float64(totalMetrics)*100)

	if *byOperation && len(byOperationMap) > 0 {
		fmt.Println("\nDuration by Operation:")
		fmt.Println("Operation               | Count | Total (ms) | Avg (ms)")
		fmt.Println("------------------------|-------|------------|----------")

		for op, totalMs := range byOperationMap {
			count := byOperationCount[op]
			avgMs := totalMs / int64(count)
			opName := truncate(op, 23)
			fmt.Printf("%-23s | %5d | %10d | %8d\n", opName, count, totalMs, avgMs)
		}
	}

	if !*byOperation {
		// Print individual metrics table
		fmt.Println("\nIndividual Metrics:")
		fmt.Println("Timestamp  | Operation               | Duration (ms) | Success | Context")
		fmt.Println("-----------|-------------------------|---------------|---------|--------------------")

		for _, m := range metrics {
			timestamp := time.Unix(m.Timestamp, 0).Format("2006-01-02")
			operation := truncate(m.Operation, 23)
			successStr := "✓"
			if !m.Success {
				successStr = "✗"
			}
			context := truncate(m.Context, 19)
			fmt.Printf("%s | %-23s | %13d | %-7s | %s\n",
				timestamp, operation, m.DurationMs, successStr, context)
		}
	}
}
