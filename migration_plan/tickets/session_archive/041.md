# GOgent-041: Intent Classification & Keyword Extraction

## Description

Extend user intent capture with automatic classification and keyword extraction. This enables:
- Filtered queries ("show me all routing preferences")
- Aggregation by category ("most common workflow requests")
- Cross-session pattern detection ("user always prefers X")

## Design Intent

Raw Q&A capture (037d) provides data but not insight. Classification transforms unstructured responses into queryable behavioral patterns. The approach is **deterministic pattern matching**, not LLM inference - keeping it fast, cheap, and consistent.

## Intended End State

- Every captured UserIntent has a `Category` field
- Keywords extracted and stored in `Keywords` array
- Classification happens at capture time (in 037d hook)
- QueryUserIntents supports Category and Keywords filters
- CLI supports `--category` flag
- 90%+ classification accuracy on common intents

## Dependencies

- GOgent-037d: User Intent Capture Hook (provides capture mechanism)

## Implementation Details

### Category Taxonomy

| Category | Trigger Patterns (in question OR response) |
|----------|-------------------------------------------|
| `routing` | tier, model, agent, delegate, sonnet, haiku, opus, use .* for |
| `tooling` | tool, edit, bash, grep, use .* not, prefer .* over, don't use |
| `style` | concise, verbose, format, output, shorter, longer, response |
| `workflow` | always, never, after .* do, before .* do, sequence, order, first .* then |
| `domain` | we use, our project, codebase, convention, framework, library |
| `correction` | no,? (I\|you) meant, actually, not that, wrong, incorrect, other |
| `approval` | yes, proceed, go ahead, looks good, confirmed, approve, do it |
| `rejection` | stop, cancel, abort, wrong approach, undo, don't, halt |
| `general` | (default when no pattern matches) |

### Classification Implementation

**File: `pkg/session/intent_classifier.go`** (NEW)

```go
package session

import (
    "regexp"
    "strings"
)

// IntentCategory represents the type of user intent
type IntentCategory string

const (
    CategoryRouting    IntentCategory = "routing"
    CategoryTooling    IntentCategory = "tooling"
    CategoryStyle      IntentCategory = "style"
    CategoryWorkflow   IntentCategory = "workflow"
    CategoryDomain     IntentCategory = "domain"
    CategoryCorrection IntentCategory = "correction"
    CategoryApproval   IntentCategory = "approval"
    CategoryRejection  IntentCategory = "rejection"
    CategoryGeneral    IntentCategory = "general"
)

// categoryPatterns maps categories to regex patterns
var categoryPatterns = map[IntentCategory]*regexp.Regexp{
    CategoryRouting:    regexp.MustCompile(`(?i)\b(tier|model|agent|delegate|sonnet|haiku|opus|use\s+\w+\s+for)\b`),
    CategoryTooling:    regexp.MustCompile(`(?i)\b(tool|edit|bash|grep|glob|read|write|use\s+\w+\s+not|prefer\s+\w+\s+over|don'?t\s+use)\b`),
    CategoryStyle:      regexp.MustCompile(`(?i)\b(concise|verbose|format|output|shorter|longer|brief|detailed|response\s+style)\b`),
    CategoryWorkflow:   regexp.MustCompile(`(?i)\b(always|never|after\s+\w+\s+do|before\s+\w+\s+do|sequence|order|first\s+\w+\s+then)\b`),
    CategoryDomain:     regexp.MustCompile(`(?i)\b(we\s+use|our\s+project|codebase|convention|framework|library|this\s+project)\b`),
    CategoryCorrection: regexp.MustCompile(`(?i)\b(no,?\s*(I|you)\s+meant|actually|not\s+that|wrong|incorrect|the\s+other)\b`),
    CategoryApproval:   regexp.MustCompile(`(?i)^(yes|proceed|go\s+ahead|looks\s+good|confirmed|approve|do\s+it|ok|okay|sure|yep|yeah)\b`),
    CategoryRejection:  regexp.MustCompile(`(?i)^(stop|cancel|abort|wrong\s+approach|undo|don'?t|halt|no\s+don'?t|wait)\b`),
}

// ClassifyIntent determines the category of a user intent
// Uses pattern matching on combined question + response text
func ClassifyIntent(question, response string) IntentCategory {
    combined := strings.ToLower(question + " " + response)

    // Check patterns in priority order
    // Correction/rejection/approval checked first (strong signals)
    priorityOrder := []IntentCategory{
        CategoryCorrection,
        CategoryRejection,
        CategoryApproval,
        CategoryRouting,
        CategoryTooling,
        CategoryWorkflow,
        CategoryDomain,
        CategoryStyle,
    }

    for _, category := range priorityOrder {
        if pattern, ok := categoryPatterns[category]; ok {
            if pattern.MatchString(combined) {
                return category
            }
        }
    }

    return CategoryGeneral
}
```

### Keyword Extraction Implementation

**File: `pkg/session/keyword_extractor.go`** (NEW)

```go
package session

import (
    "path/filepath"
    "regexp"
    "strings"
)

// Known keywords to extract
var knownKeywords = map[string]bool{
    // Tools
    "edit": true, "bash": true, "read": true, "write": true,
    "glob": true, "grep": true, "task": true, "webfetch": true,
    // Models
    "sonnet": true, "haiku": true, "opus": true,
    // Test frameworks
    "pytest": true, "jest": true, "mocha": true, "unittest": true,
    "go test": true, "rspec": true, "testthat": true,
    // Build tools
    "make": true, "npm": true, "yarn": true, "cargo": true,
    "go build": true, "pip": true, "uv": true,
    // VCS
    "git": true, "commit": true, "push": true, "pull": true,
}

// filePathPattern matches common file paths
var filePathPattern = regexp.MustCompile(`[\w./]+\.(go|py|js|ts|rs|r|md|json|yaml|yml|toml)`)

// ExtractKeywords extracts relevant keywords from a response
// Returns at most 10 keywords, deduplicated
func ExtractKeywords(response string) []string {
    lower := strings.ToLower(response)
    seen := make(map[string]bool)
    var keywords []string

    // Extract known keywords
    for keyword := range knownKeywords {
        if strings.Contains(lower, keyword) {
            if !seen[keyword] {
                seen[keyword] = true
                keywords = append(keywords, keyword)
            }
        }
    }

    // Extract file paths
    matches := filePathPattern.FindAllString(response, -1)
    for _, match := range matches {
        // Use just the filename, not full path
        name := filepath.Base(match)
        if !seen[name] {
            seen[name] = true
            keywords = append(keywords, name)
        }
    }

    // Limit to 10 keywords
    if len(keywords) > 10 {
        keywords = keywords[:10]
    }

    return keywords
}
```

func TestClassifyIntent_Routing(t *testing.T) {
    cases := []struct {
        question string
        response string
        expected IntentCategory
    }{
        {"Which model?", "Use sonnet for this", CategoryRouting},
        {"How to proceed?", "Delegate to haiku", CategoryRouting},
        {"", "use opus", CategoryRouting},
    }
    for _, tc := range cases {
        result := ClassifyIntent(tc.question, tc.response)
        if result != tc.expected {
            t.Errorf("ClassifyIntent(%q, %q) = %s, want %s",
                tc.question, tc.response, result, tc.expected)
        }
    }
}

func TestClassifyIntent_Correction(t *testing.T) {
    // Correction should take priority
    result := ClassifyIntent("Which file?", "No, I meant the other one")
    if result != CategoryCorrection {
        t.Errorf("Expected correction, got %s", result)
    }
}

func TestClassifyIntent_Approval(t *testing.T) {
    cases := []string{"yes", "Yes", "proceed", "Go ahead", "looks good", "do it"}
    for _, response := range cases {
        result := ClassifyIntent("Ready?", response)
        if result != CategoryApproval {
            t.Errorf("ClassifyIntent(_, %q) = %s, want approval", response, result)
        }
    }
}
