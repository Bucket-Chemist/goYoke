package routing

import (
	"fmt"
	"time"
)

// min returns the smaller of two integers
func min(a, b int) int {
	if a < b {
		return a
	}
	return b
}

// GenerateResumeGuidance creates prioritized next steps from session context
func GenerateResumeGuidance(wip *WIPContext, learnings []string, violations []string) []string {
	var guidance []string

	// Priority 1: Resume point from WIP
	if wip != nil {
		resumeMsg := fmt.Sprintf("**RESUME POINT**: %s - Last edited %s ago",
			wip.FilePath,
			formatTimeSince(wip.Timestamp))

		if wip.LineContext != "" {
			resumeMsg += fmt.Sprintf("\n  Context: `%s`", wip.LineContext[:min(60, len(wip.LineContext))])
		}

		guidance = append(guidance, resumeMsg)
	}

	// Priority 2: Sharp edges requiring attention
	if len(learnings) > 0 {
		guidance = append(guidance, fmt.Sprintf("**SHARP EDGES**: %d debugging loops captured - review before continuing", len(learnings)))
		for i, learning := range learnings {
			if i >= 3 {
				guidance = append(guidance, fmt.Sprintf("  ... and %d more", len(learnings)-3))
				break
			}
			guidance = append(guidance, "  "+learning)
		}
	}

	// Priority 3: Routing violations patterns
	if len(violations) > 0 {
		guidance = append(guidance, fmt.Sprintf("**VIOLATIONS**: %d routing issues detected - review patterns", len(violations)))
		// Show first 2
		for i, violation := range violations {
			if i >= 2 {
				guidance = append(guidance, fmt.Sprintf("  ... and %d more", len(violations)-2))
				break
			}
			guidance = append(guidance, "  "+violation)
		}
	}

	return guidance
}

// formatTimeSince converts timestamp to human-readable elapsed time
func formatTimeSince(timestamp int64) string {
	elapsed := time.Now().Unix() - timestamp
	if elapsed < 60 {
		return fmt.Sprintf("%ds", elapsed)
	} else if elapsed < 3600 {
		return fmt.Sprintf("%dm", elapsed/60)
	} else {
		return fmt.Sprintf("%dh", elapsed/3600)
	}
}

	wip := &WIPContext{
		FilePath:    "test.go",
		LineContext: "func main() {",
		Timestamp:   time.Now().Unix() - 300, // 5 minutes ago
		ToolUsed:    "Edit",
	}

	result := GenerateResumeGuidance(wip, nil, nil)

	if len(result) != 1 {
		t.Errorf("Expected 1 guidance item, got: %d", len(result))

		t.Error("Expected file path in guidance")
