func main() {
	// Parse flags
	forceFlag := flag.Bool("force", false, "Force aggregation regardless of file size")
	dryRunFlag := flag.Bool("dry-run", false, "Show what would be aggregated without doing it")
	helpFlag := flag.Bool("help", false, "Show help message")
	flag.Parse()

	if *helpFlag {
		printHelp()
		return
	}

	// Get project directory
	projectDir := os.Getenv("GOGENT_PROJECT_DIR")
	if projectDir == "" {
		cwd, err := os.Getwd()
		if err != nil {
			fmt.Fprintf(os.Stderr, "[gogent-aggregate] Failed to get working directory: %v\n", err)
			fmt.Fprintln(os.Stderr, "  Set GOGENT_PROJECT_DIR environment variable or run from project root.")
			os.Exit(1)
		}
		projectDir = cwd
	}

	memoryDir := filepath.Join(projectDir, ".claude", "memory")
	archiveDir := filepath.Join(memoryDir, "archive")

	// Define JSONL files to check (complete list of all learning artifacts)
	files := []string{
		filepath.Join(memoryDir, "pending-learnings.jsonl"),   // SharpEdges (029)
		filepath.Join(memoryDir, "user-intents.jsonl"),        // UserIntents (029a) - NEW
		filepath.Join(memoryDir, "decisions.jsonl"),           // Decisions (029c)
		filepath.Join(memoryDir, "preferences.jsonl"),         // Preferences (029c)
		filepath.Join(memoryDir, "performance.jsonl"),         // Performance (029c)
		filepath.Join(memoryDir, "routing-violations.jsonl"),  // Existing
	}

	// Check which files need aggregation
	needsAggregation := false
	for _, file := range files {
		size, err := getFileSize(file)
		if err != nil && !os.IsNotExist(err) {
			fmt.Fprintf(os.Stderr, "[gogent-aggregate] Warning: Failed to stat %s: %v\n", file, err)
			continue
		}

		if size > MaxFileSizeBytes || *forceFlag {
			needsAggregation = true
			if *dryRunFlag {
				fmt.Printf("[dry-run] Would aggregate %s (size: %d bytes)\n", filepath.Base(file), size)
			}
		}
	}

	if !needsAggregation {
		fmt.Println("No aggregation needed (all files < 1MB)")
		return
	}

	if *dryRunFlag {
		fmt.Println("\n[dry-run] No files were modified")
		return
	}

	// Ensure archive directory exists
	if err := os.MkdirAll(archiveDir, 0755); err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-aggregate] Failed to create archive directory: %v\n", err)
		os.Exit(1)
	}

	// Perform aggregation
	week := time.Now().Format(WeekFormat)
	stats := aggregateFiles(files, archiveDir, week)

	// Generate summary
	summaryPath := filepath.Join(archiveDir, week+"-summary.md")
	if err := generateSummary(summaryPath, week, stats); err != nil {
		fmt.Fprintf(os.Stderr, "[gogent-aggregate] Failed to generate summary: %v\n", err)
		os.Exit(1)
	}

	fmt.Printf("Aggregation complete: %s\n", week)
	fmt.Printf("  Sharp Edges: %d\n", stats.SharpEdges)
	fmt.Printf("  User Intents: %d\n", stats.UserIntents)  // NEW
	fmt.Printf("  Decisions: %d\n", stats.Decisions)
	fmt.Printf("  Preferences: %d\n", stats.Preferences)
	fmt.Printf("  Performance: %d\n", stats.Performance)
	fmt.Printf("  Violations: %d\n", stats.Violations)
	fmt.Printf("  Summary: %s\n", summaryPath)
}

// getFileSize returns file size in bytes, 0 if file doesn't exist
func getFileSize(path string) (int64, error) {
	info, err := os.Stat(path)
	if err != nil {
		return 0, err
	}
	return info.Size(), nil
}

// AggregationStats tracks counts of archived artifacts
type AggregationStats struct {
	SharpEdges  int
	UserIntents int  // NEW
	Decisions   int
	Preferences int
	Performance int
	Violations  int
}

// aggregateFiles moves JSONL files to archive with week prefix
func aggregateFiles(files []string, archiveDir, week string) AggregationStats {
	stats := AggregationStats{}

	for _, srcPath := range files {
		if _, err := os.Stat(srcPath); os.IsNotExist(err) {
			continue // Skip missing files
		}

		// Determine file type from name
		basename := filepath.Base(srcPath)
		var fileType string
		switch basename {
		case "pending-learnings.jsonl":
			fileType = "sharp-edges"
		case "user-intents.jsonl":
			fileType = "user-intents"  // NEW
		case "decisions.jsonl":
			fileType = "decisions"
		case "preferences.jsonl":
			fileType = "preferences"
		case "performance.jsonl":
			fileType = "performance"
		case "routing-violations.jsonl":
			fileType = "violations"
		default:
			continue
		}

		// Archive path
		archivePath := filepath.Join(archiveDir, week+"-"+fileType+".jsonl")

		// Count lines
		count, err := countLines(srcPath)
		if err != nil {
			fmt.Fprintf(os.Stderr, "[gogent-aggregate] Warning: Failed to count lines in %s: %v\n", srcPath, err)
			count = 0
		}

		// Update stats
		switch fileType {
		case "sharp-edges":
			stats.SharpEdges = count
		case "user-intents":
			stats.UserIntents = count  // NEW
		case "decisions":
			stats.Decisions = count
		case "preferences":
			stats.Preferences = count
		case "performance":
			stats.Performance = count
		case "violations":
			stats.Violations = count
		}

		// Move file to archive
		if err := os.Rename(srcPath, archivePath); err != nil {
			fmt.Fprintf(os.Stderr, "[gogent-aggregate] Warning: Failed to archive %s: %v\n", srcPath, err)
			continue
		}

		// Create empty replacement file
		if err := os.WriteFile(srcPath, []byte(""), 0644); err != nil {
			fmt.Fprintf(os.Stderr, "[gogent-aggregate] Warning: Failed to create replacement %s: %v\n", srcPath, err)
		}
	}

	return stats
}

// countLines counts non-empty lines in a file
func countLines(path string) (int, error) {
	file, err := os.Open(path)
	if err != nil {
		return 0, err
	}
	defer file.Close()

	count := 0
	scanner := bufio.NewScanner(file)
	for scanner.Scan() {
		line := strings.TrimSpace(scanner.Text())
		if line != "" {
			count++
		}
	}

	if err := scanner.Err(); err != nil {
		return 0, err
	}

	return count, nil
}

// generateSummary creates a markdown summary of the archived week
func generateSummary(summaryPath, week string, stats AggregationStats) error {
	summary := fmt.Sprintf(`# Weekly Learning Summary - %s

## Overview

Generated: %s
Week: %s

## Artifact Counts

- **Sharp Edges**: %d
- **User Intents**: %d
- **Decisions**: %d
- **Preferences**: %d
- **Performance Metrics**: %d
- **Routing Violations**: %d

## Archive Location

All artifacts for this week are archived in:
- sharp edges: `.claude/memory/archive/%s-sharp-edges.jsonl`
- user intents: `.claude/memory/archive/%s-user-intents.jsonl`
- decisions: `.claude/memory/archive/%s-decisions.jsonl`
- preferences: `.claude/memory/archive/%s-preferences.jsonl`
- performance: `.claude/memory/archive/%s-performance.jsonl`
- violations: `.claude/memory/archive/%s-violations.jsonl`

## Usage

Query archived decisions:
` + "```bash" + `
# Example: Find architectural decisions from this week
grep '"category":"architecture"' .claude/memory/archive/%s-decisions.jsonl
` + "```" + `

Review sharp edges:
` + "```bash" + `
# Example: Count error types
jq '.error_type' .claude/memory/archive/%s-sharp-edges.jsonl | sort | uniq -c
` + "```" + `

---

**Note**: This is an automated weekly aggregation. Original JSONL files have been archived and reset.
`,
		week,
		time.Now().Format("2006-01-02 15:04:05"),
		week,
		stats.SharpEdges,
		stats.UserIntents,
		stats.Decisions,
		stats.Preferences,
		stats.Performance,
		stats.Violations,
		week, week, week, week, week, week,  // 6 archive paths now
		week,
		week,
	)

	if err := os.WriteFile(summaryPath, []byte(summary), 0644); err != nil {
		return fmt.Errorf("failed to write summary: %w", err)
	}

	return nil
}
